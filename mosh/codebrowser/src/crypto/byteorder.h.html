61/* Use our fallback implementation, which is correct for any endianness. */

71/* Use unions rather than casts, to comply with strict aliasing rules. */
72
73inline uint64_t htobe64( uint64_t x ) {
74 uint8_t xs[ 8 ] = {
75 static_cast<uint8_t>( ( x >> 56 ) & 0xFF ),
76 static_cast<uint8_t>( ( x >> 48 ) & 0xFF ),
77 static_cast<uint8_t>( ( x >> 40 ) & 0xFF ),
78 static_cast<uint8_t>( ( x >> 32 ) & 0xFF ),
79 static_cast<uint8_t>( ( x >> 24 ) & 0xFF ),
80 static_cast<uint8_t>( ( x >> 16 ) & 0xFF ),
81 static_cast<uint8_t>( ( x >> 8 ) & 0xFF ),
82 static_cast<uint8_t>( ( x ) & 0xFF ) };
83 union {
84 const uint8_t *p8;
85 const uint64_t *p64;
86 } u;
87 u.p8 = xs;
88 return *u.p64;
89}
90
91inline uint64_t be64toh( uint64_t x ) {
92 union {
93 const uint8_t *p8;
94 const uint64_t *p64;
95 } u;
96 u.p64 = &x;
97 return ( uint64_t( u.p8[ 0 ] ) << 56 )
98 | ( uint64_t( u.p8[ 1 ] ) << 48 )
99 | ( uint64_t( u.p8[ 2 ] ) << 40 )
100 | ( uint64_t( u.p8[ 3 ] ) << 32 )
101 | ( uint64_t( u.p8[ 4 ] ) << 24 )
102 | ( uint64_t( u.p8[ 5 ] ) << 16 )
103 | ( uint64_t( u.p8[ 6 ] ) << 8 )
104 | ( uint64_t( u.p8[ 7 ] ) );
105}
106
107inline uint16_t htobe16( uint16_t x ) {
108 uint8_t xs[ 2 ] = {
109 static_cast<uint8_t>( ( x >> 8 ) & 0xFF ),
110 static_cast<uint8_t>( ( x ) & 0xFF ) };
111 union {
112 const uint8_t *p8;
113 const uint16_t *p16;
114 } u;
115 u.p8 = xs;
116 return *u.p16;
117}
118
119inline uint16_t be16toh( uint16_t x ) {
120 union {
121 const uint8_t *p8;
122 const uint16_t *p16;
123 } u;
124 u.p16 = &x;
125 return ( uint16_t( u.p8[ 0 ] ) << 8 )
126 | ( uint16_t( u.p8[ 1 ] ) );
127}
