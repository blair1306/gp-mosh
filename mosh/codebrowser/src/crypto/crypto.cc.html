50long int myatoi( const char *str )
51{
52 char *end;
53
54 errno = 0;
55 long int ret = strtol( str, &end, 10 );
56
57 if ( ( errno != 0 )
58 || ( end != str + strlen( str ) ) ) {
59 throw CryptoException( "Bad integer." );
60 }
61
62 return ret;
63}

65uint64_t Crypto::unique( void )
66{
67 static uint64_t counter = 0;
68 uint64_t rv = counter++;
69 if ( counter == 0 ) {
70 throw CryptoException( "Counter wrapped", true );
71 }
72 return rv;
73}

75AlignedBuffer::AlignedBuffer( size_t len, const char *data )
76 : m_len( len ), m_allocated( NULL ), m_data( NULL )
77{
78 size_t alloc_len = len ? len : 1;
79#if defined(HAVE_POSIX_MEMALIGN)
80 if ( ( 0 != posix_memalign( &m_allocated, 16, alloc_len ) )
81 || ( m_allocated == NULL ) ) {
82 throw std::bad_alloc();
83 }
84 m_data = (char *) m_allocated;
85
86#else
87 /* malloc() a region 15 bytes larger than we need, and find
88 the aligned offset within. */
89 m_allocated = malloc( 15 + alloc_len );
90 if ( m_allocated == NULL ) {
91 throw std::bad_alloc();
92 }
93
94 uintptr_t iptr = (uintptr_t) m_allocated;
95 if ( iptr & 0xF ) {
96 iptr += 16 - ( iptr & 0xF );
97 }
98 assert( !( iptr & 0xF ) );
99 assert( iptr >= (uintptr_t) m_allocated );
100 assert( iptr <= ( 15 + (uintptr_t) m_allocated ) );
101
102 m_data = (char *) iptr;
103
104#endif /* !defined(HAVE_POSIX_MEMALIGN) */
105
106 if ( data ) {
107 memcpy( m_data, data, len );
108 }
109}

111Base64Key::Base64Key( string printable_key )
112{
113 if ( printable_key.length() != 22 ) {
114 throw CryptoException( "Key must be 22 letters long." );
115 }
116
117 string base64 = printable_key + "==";
118
119 size_t len = 16;
120 if ( !base64_decode( base64.data(), 24, key, &len ) ) {
121 throw CryptoException( "Key must be well-formed base64." );
122 }
123
124 if ( len != 16 ) {
125 throw CryptoException( "Key must represent 16 octets." );
126 }
127
128 /* to catch changes after the first 128 bits */
129 if ( printable_key != this->printable_key() ) {
130 throw CryptoException( "Base64 key was not encoded 128-bit key." );
131 }
132}

144string Base64Key::printable_key( void ) const
145{
146 char base64[ 24 ];
147
148 base64_encode( key, 16, base64, 24 );
149
150 if ( (base64[ 23 ] != '=')
151 || (base64[ 22 ] != '=') ) {
152 throw CryptoException( string( "Unexpected output from base64_encode: " ) + string( base64, 24 ) );
153 }
154
155 base64[ 22 ] = 0;
156 return string( base64 );
157}

159Session::Session( Base64Key s_key )
160 : key( s_key ), ctx_buf( ae_ctx_sizeof() ),
161 ctx( (ae_ctx *)ctx_buf.data() ), blocks_encrypted( 0 ),
162 plaintext_buffer( RECEIVE_MTU ),
163 ciphertext_buffer( RECEIVE_MTU ),
164 nonce_buffer( Nonce::NONCE_LEN )

176Nonce::Nonce( uint64_t val )
177{
178 uint64_t val_net = htobe64( val );
179
180 memset( bytes, 0, 4 );
181 memcpy( bytes + 4, &val_net, 8 );

184uint64_t Nonce::val( void ) const
185{
186 uint64_t ret;
187 memcpy( &ret, bytes + 4, 8 );
188 return be64toh( ret );

201const string Session::encrypt( const Message & plaintext )
202{
203 const size_t pt_len = plaintext.text.size();
204 const int ciphertext_len = pt_len + 16;
205
206 assert( (size_t)ciphertext_len <= ciphertext_buffer.len() );
207 assert( pt_len <= plaintext_buffer.len() );
208
209 memcpy( plaintext_buffer.data(), plaintext.text.data(), pt_len );
210 memcpy( nonce_buffer.data(), plaintext.nonce.data(), Nonce::NONCE_LEN );
211
212 if ( ciphertext_len != ae_encrypt( ctx, /* ctx */
213 nonce_buffer.data(), /* nonce */
214 plaintext_buffer.data(), /* pt */
215 pt_len, /* pt_len */
216 NULL, /* ad */
217 0, /* ad_len */
218 ciphertext_buffer.data(), /* ct */
219 NULL, /* tag */
220 AE_FINALIZE ) ) { /* final */
221 throw CryptoException( "ae_encrypt() returned error." );
222 }
223
224 blocks_encrypted += pt_len >> 4;
225 if ( pt_len & 0xF ) {
226 /* partial block */
227 blocks_encrypted++;
228 }
229
230 /* "Both the privacy and the authenticity properties of OCB degrade as
231 per s^2 / 2^128, where s is the total number of blocks that the
232 adversary acquires.... In order to ensure that s^2 / 2^128 remains
233 small, a given key should be used to encrypt at most 2^48 blocks (2^55
234 bits or 4 petabytes)"
235
236 -- http://tools.ietf.org/html/draft-krovetz-ocb-03
237
238 We deem it unlikely that a legitimate user will send 4 PB through a Mosh
239 session. If it happens, we simply kill the session. The server and
240 client use the same key, so we actually need to die after 2^47 blocks.
241 */
242 if ( blocks_encrypted >> 47 ) {
243 throw CryptoException( "Encrypted 2^47 blocks.", true );
244 }
245
246 string text( ciphertext_buffer.data(), ciphertext_len );
247
248 return plaintext.nonce.cc_str() + text;
249}

251const Message Session::decrypt( const char *str, size_t len )
252{
253 if ( len < 24 ) {
254 throw CryptoException( "Ciphertext must contain nonce and tag." );
255 }
256
257 int body_len = len - 8;
258 int pt_len = body_len - 16;
259
260 if ( pt_len < 0 ) { /* super-assertion that pt_len does not equal AE_INVALID */
261 fprintf( stderr, "BUG.\n" );
262 exit( 1 );
263 }
264
265 assert( (size_t)body_len <= ciphertext_buffer.len() );
266 assert( (size_t)pt_len <= plaintext_buffer.len() );
267
268 Nonce nonce( str, 8 );
269 memcpy( ciphertext_buffer.data(), str + 8, body_len );
270 memcpy( nonce_buffer.data(), nonce.data(), Nonce::NONCE_LEN );
271
272 if ( pt_len != ae_decrypt( ctx, /* ctx */
273 nonce_buffer.data(), /* nonce */
274 ciphertext_buffer.data(), /* ct */
275 body_len, /* ct_len */
276 NULL, /* ad */
277 0, /* ad_len */
278 plaintext_buffer.data(), /* pt */
279 NULL, /* tag */
280 AE_FINALIZE ) ) { /* final */
281 throw CryptoException( "Packet failed integrity check." );
282 }
283
284 const Message ret( nonce, string( plaintext_buffer.data(), pt_len ) );
285
286 return ret;
287}

289static rlim_t saved_core_rlimit;
290
291/* Disable dumping core, as a precaution to avoid saving sensitive data
292 to disk. */
293void Crypto::disable_dumping_core( void ) {
294 struct rlimit limit;
295 if ( 0 != getrlimit( RLIMIT_CORE, &limit ) ) {
296 /* We don't throw CryptoException because this is called very early
297 in main(), outside of 'try'. */
298 perror( "getrlimit(RLIMIT_CORE)" );
299 exit( 1 );
300 }
301
302 saved_core_rlimit = limit.rlim_cur;
303 limit.rlim_cur = 0;
304 if ( 0 != setrlimit( RLIMIT_CORE, &limit ) ) {
305 perror( "setrlimit(RLIMIT_CORE)" );
306 exit( 1 );
307 }
308}
309
310void Crypto::reenable_dumping_core( void ) {
311 /* Silent failure is safe. */
312 struct rlimit limit;
313 if ( 0 == getrlimit( RLIMIT_CORE, &limit ) ) {
314 limit.rlim_cur = saved_core_rlimit;
315 setrlimit( RLIMIT_CORE, &limit );
316 }
