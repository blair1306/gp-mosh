46void ConditionalOverlayCell::apply( Framebuffer &fb, uint64_t confirmed_epoch, int row, bool flag ) const
47{
48 if ( (!active)
49 || (row >= fb.ds.get_height())
50 || (col >= fb.ds.get_width()) ) {
51 return;
52 }
53
54 if ( tentative( confirmed_epoch ) ) {
55 return;
56 }
57
58 if ( replacement.is_blank() && fb.get_cell( row, col )->is_blank() ) {
59 flag = false;
60 }
61
62 if ( unknown ) {
63 if ( flag && ( col != fb.ds.get_width() - 1 ) ) {
64 fb.get_mutable_cell( row, col )->get_renditions().set_attribute(Renditions::underlined, true);
65 }
66 return;
67 }
68
69 if ( *fb.get_cell( row, col ) != replacement ) {
70 *(fb.get_mutable_cell( row, col )) = replacement;
71 if ( flag ) {
72 fb.get_mutable_cell( row, col )->get_renditions().set_attribute( Renditions::underlined, true );
73 }
74 }
75}

77Validity ConditionalOverlayCell::get_validity( const Framebuffer &fb, int row,
78 uint64_t early_ack __attribute__((unused)),
79 uint64_t late_ack ) const
80{
81 if ( !active ) {
82 return Inactive;
83 }
84
85 if ( (row >= fb.ds.get_height())
86 || (col >= fb.ds.get_width()) ) {
87 return IncorrectOrExpired;
88 }
89
90 const Cell &current = *( fb.get_cell( row, col ) );
91
92 /* see if it hasn't been updated yet */
93 if ( late_ack < expiration_frame ) {
94 return Pending;
95 }
96
97 if ( unknown ) {
98 return CorrectNoCredit;
99 }
100
101 if ( replacement.is_blank() ) { /* too easy for this to trigger falsely */
102 return CorrectNoCredit;
103 }
104
105 if ( current.contents_match( replacement ) ) {
106 vector<Cell>::const_iterator it = original_contents.begin();
107 for ( ; it != original_contents.end(); it++ ) {
108 if ( it->contents_match( replacement ) )
109 break;
110 }
111 if ( it == original_contents.end() ) {
112 return Correct;
113 }
114 return CorrectNoCredit;
115 }
116 return IncorrectOrExpired;
117}

119Validity ConditionalCursorMove::get_validity( const Framebuffer &fb,
120 uint64_t early_ack __attribute((unused)),
121 uint64_t late_ack ) const
122{
123 if ( !active ) {
124 return Inactive;
125 }
126
127 if ( (row >= fb.ds.get_height())
128 || (col >= fb.ds.get_width()) ) {
129 // fprintf( stderr, "Crazy cursor (%d,%d)!\n", row, col );
130 return IncorrectOrExpired;
131 }
132
133 if ( late_ack >= expiration_frame ) {
134 if ( (fb.ds.get_cursor_col() == col)
135 && (fb.ds.get_cursor_row() == row) ) {
136 return Correct;
137 }
138 return IncorrectOrExpired;
139 }
140
141 return Pending;
142}

144void ConditionalCursorMove::apply( Framebuffer &fb, uint64_t confirmed_epoch ) const
145{
146 if ( !active ) {
147 return;
148 }
149
150 if ( tentative( confirmed_epoch ) ) {
151 return;
152 }
153
154 assert( row < fb.ds.get_height() );
155 assert( col < fb.ds.get_width() );
156 assert( !fb.ds.origin_mode );
157
158 fb.ds.move_row( row, false );
159 fb.ds.move_col( col, false, false );

162NotificationEngine::NotificationEngine()
163 : last_word_from_server( timestamp() ),
164 last_acked_state( timestamp() ),
165 escape_key_string(),
166 message(),
167 message_is_network_error( false ),
168 message_expiration( -1 ),
169 show_quit_keystroke( true )
170{}

172static std::string human_readable_duration( int num_seconds, const std::string &seconds_abbr ) {
173 char tmp[ 128 ];
174 if ( num_seconds < 60 ) {
175 snprintf( tmp, 128, "%d %s", num_seconds, seconds_abbr.c_str() );
176 } else if ( num_seconds < 3600 ) {
177 snprintf( tmp, 128, "%d:%02d", num_seconds / 60, num_seconds % 60 );
178 } else {
179 snprintf( tmp, 128, "%d:%02d:%02d", num_seconds / 3600,
180 (num_seconds / 60) % 60, num_seconds % 60 );
181 }
182 return tmp;
183}

185void NotificationEngine::apply( Framebuffer &fb ) const
186{
187 uint64_t now = timestamp();
188
189 bool time_expired = need_countup( now );
190
191 if ( message.empty() && !time_expired ) {
192 return;
193 }
194
195 assert( fb.ds.get_width() > 0 );
196 assert( fb.ds.get_height() > 0 );
197
198 /* hide cursor if necessary */
199 if ( fb.ds.get_cursor_row() == 0 ) {
200 fb.ds.cursor_visible = false;
201 }
202
203 /* draw bar across top of screen */
204 Cell notification_bar( 0 );
205 notification_bar.get_renditions().foreground_color = 37;
206 notification_bar.get_renditions().background_color = 44;
207 notification_bar.append( 0x20 );
208
209 for ( int i = 0; i < fb.ds.get_width(); i++ ) {
210 *(fb.get_mutable_cell( 0, i )) = notification_bar;
211 }
212
213 /* write message */
214 wchar_t tmp[ 128 ];
215
216 /* We want to prefer the "last contact" message if we simply haven't
217 heard from the server in a while, but print the "last reply" message
218 if the problem is uplink-only. */
219
220 double since_heard = (double)(now - last_word_from_server) / 1000.0;
221 double since_ack = (double)(now - last_acked_state) / 1000.0;
222 const char server_message[] = "contact";
223 const char reply_message[] = "reply";
224
225 double time_elapsed = since_heard;
226 const char *explanation = server_message;
227
228 if ( reply_late( now ) && (!server_late( now )) ) {
229 time_elapsed = since_ack;
230 explanation = reply_message;
231 }
232
233 const static char blank[] = "";
234
235 const char *keystroke_str = show_quit_keystroke ? escape_key_string.c_str() : blank;
236
237 if ( message.empty() && (!time_expired) ) {
238 return;
239 }
240 if ( message.empty() && time_expired ) {
241 swprintf( tmp, 128, L"mosh: Last %s %s ago.%s", explanation,
242 human_readable_duration( static_cast<int>( time_elapsed ),
243 "seconds" ).c_str(),
244 keystroke_str );
245 } else if ( (!message.empty()) && (!time_expired) ) {
246 swprintf( tmp, 128, L"mosh: %ls%s", message.c_str(), keystroke_str );
247 } else {
248 swprintf( tmp, 128, L"mosh: %ls (%s without %s.)%s", message.c_str(),
249 human_readable_duration( static_cast<int>( time_elapsed ),
250 "s" ).c_str(),
251 explanation, keystroke_str );
252 }
253
254 wstring string_to_draw( tmp );
255
256 int overlay_col = 0;
257
258 Cell *combining_cell = fb.get_mutable_cell( 0, 0 );
259
260 /* We unfortunately duplicate the terminal's logic for how to render a Unicode sequence into graphemes */
261 for ( wstring::const_iterator i = string_to_draw.begin(); i != string_to_draw.end(); i++ ) {
262 if ( overlay_col >= fb.ds.get_width() ) {
263 break;
264 }
265
266 wchar_t ch = *i;
267 int chwidth = ch == L'\0' ? -1 : wcwidth( ch );
268 Cell *this_cell = 0;
269
270 switch ( chwidth ) {
271 case 1: /* normal character */
272 case 2: /* wide character */
273 this_cell = fb.get_mutable_cell( 0, overlay_col );
274 fb.reset_cell( this_cell );
275 this_cell->get_renditions().set_attribute(Renditions::bold, true);
276 this_cell->get_renditions().foreground_color = 37;
277 this_cell->get_renditions().background_color = 44;
278
279 this_cell->append( ch );
280 this_cell->set_wide( chwidth == 2 );
281 combining_cell = this_cell;
282
283 overlay_col += chwidth;
284 break;
285 case 0: /* combining character */
286 if ( !combining_cell ) {
287 break;
288 }
289
290 if ( combining_cell->empty() ) {
291 assert( !combining_cell->get_wide() );
292 combining_cell->set_fallback( true );
293 overlay_col++;
294 }
295
296 if ( !combining_cell->full() ) {
297 combining_cell->append( ch );
298 }
299 break;
300 case -1: /* unprintable character */
301 break;
302 default:
303 assert( !"unexpected character width from wcwidth()" );
304 }
305 }
306}

315int NotificationEngine::wait_time( void ) const
316{
317 uint64_t next_expiry = INT_MAX;
318
319 uint64_t now = timestamp();
320
321 next_expiry = std::min( next_expiry, message_expiration - now );
322
323 if ( need_countup( now ) ) {
324 uint64_t countup_interval = 1000;
325 if ( ( now - last_word_from_server ) > 60000 ) {
326 /* If we've been disconnected for 60 seconds, save power by updating the
327 display less often. See #243. */
328 countup_interval = Network::ACK_INTERVAL;
329 }
330 next_expiry = std::min( next_expiry, countup_interval );
331 }
332
333 return next_expiry;
334}

359void PredictionEngine::apply( Framebuffer &fb ) const
360{
361 if ( (display_preference == Never) || !( srtt_trigger
362 || glitch_trigger
363 || (display_preference == Always)
364 || (display_preference == Experimental) ) ){
365 return;
366 }
367
368 for ( cursors_type::const_iterator it = cursors.begin();
369 it != cursors.end();
370 it++ ) {
371 it->apply( fb, confirmed_epoch );
372 }
373
374 for ( overlays_type::const_iterator it = overlays.begin();
375 it != overlays.end();
376 it++ ) {
377 it->apply( fb, confirmed_epoch, flagging );
378 }

381void PredictionEngine::kill_epoch( uint64_t epoch, const Framebuffer &fb )
382{
383 cursors.remove_if( bind2nd( mem_fun_ref( &ConditionalCursorMove::tentative ), epoch - 1 ) );
384
385 cursors.push_back( ConditionalCursorMove( local_frame_sent + 1,
386 fb.ds.get_cursor_row(),
387 fb.ds.get_cursor_col(),
388 prediction_epoch ) );
389 cursor().active = true;
390
391 for ( overlays_type::iterator i = overlays.begin();
392 i != overlays.end();
393 i++ ) {
394 for ( overlay_cells_type::iterator j = i->overlay_cells.begin();
395 j != i->overlay_cells.end();
396 j++ ) {
397 if ( j->tentative( epoch - 1 ) ) {
398 j->reset();
399 }
400 }
401 }
402
403 become_tentative();
404}

415void PredictionEngine::init_cursor( const Framebuffer &fb )
416{
417 if ( cursors.empty() ) {
418 /* initialize new cursor prediction */
419
420 cursors.push_back( ConditionalCursorMove( local_frame_sent + 1,
421 fb.ds.get_cursor_row(),
422 fb.ds.get_cursor_col(),
423 prediction_epoch ) );
424
425 cursor().active = true;
426 } else if ( cursor().tentative_until_epoch != prediction_epoch ) {
427 cursors.push_back( ConditionalCursorMove( local_frame_sent + 1,
428 cursor().row,
429 cursor().col,
430 prediction_epoch ) );
431
432 cursor().active = true;
433 }

436void PredictionEngine::cull( const Framebuffer &fb )
437{
438 if ( display_preference == Never ) {
439 return;
440 }
441
442 if ( (last_height != fb.ds.get_height())
443 || (last_width != fb.ds.get_width()) ) {
444 last_height = fb.ds.get_height();
445 last_width = fb.ds.get_width();
446 reset();
447 }
448
449 uint64_t now = timestamp();
450
451 /* control srtt_trigger with hysteresis */
452 if ( send_interval > SRTT_TRIGGER_HIGH ) {
453 srtt_trigger = true;
454 } else if ( srtt_trigger &&
455 (send_interval <= SRTT_TRIGGER_LOW) /* 20 ms is current minimum value */
456 && (!active()) ) { /* only turn off when no predictions being shown */
457 srtt_trigger = false;
458 }
459
460 /* control underlining with hysteresis */
461 if ( send_interval > FLAG_TRIGGER_HIGH ) {
462 flagging = true;
463 } else if ( send_interval <= FLAG_TRIGGER_LOW ) {
464 flagging = false;
465 }
466
467 /* really big glitches also activate underlining */
468 if ( glitch_trigger > GLITCH_REPAIR_COUNT ) {
469 flagging = true;
470 }
471
472 /* go through cell predictions */
473
474 overlays_type::iterator i = overlays.begin();
475 while ( i != overlays.end() ) {
476 overlays_type::iterator inext = i;
477 inext++;
478 if ( (i->row_num < 0) || (i->row_num >= fb.ds.get_height()) ) {
479 overlays.erase( i );
480 i = inext;
481 continue;
482 }
483
484 for ( overlay_cells_type::iterator j = i->overlay_cells.begin();
485 j != i->overlay_cells.end();
486 j++ ) {
487 switch ( j->get_validity( fb, i->row_num,
488 local_frame_acked, local_frame_late_acked ) ) {
489 case IncorrectOrExpired:
490 if ( j->tentative( confirmed_epoch ) ) {
491
492 /*
493 fprintf( stderr, "Bad tentative prediction in row %d, col %d (think %lc, actually %lc)\n",
494 i->row_num, j->col,
495 j->replacement.debug_contents(),
496 fb.get_cell( i->row_num, j->col )->debug_contents()
497 );
498 */
499
500 if ( display_preference == Experimental ) {
501 j->reset();
502 } else {
503 kill_epoch( j->tentative_until_epoch, fb );
504 }
505 /*
506 if ( j->display_time != uint64_t(-1) ) {
507 fprintf( stderr, "TIMING %ld - %ld (TENT)\n", time(NULL), now - j->display_time );
508 }
509 */
510 } else {
511 /*
512 fprintf( stderr, "[%d=>%d] Killing prediction in row %d, col %d (think %lc, actually %lc)\n",
513 (int)local_frame_acked, (int)j->expiration_frame,
514 i->row_num, j->col,
515 j->replacement.debug_contents(),
516 fb.get_cell( i->row_num, j->col )->debug_contents() );
517 */
518 /*
519 if ( j->display_time != uint64_t(-1) ) {
520 fprintf( stderr, "TIMING %ld - %ld\n", time(NULL), now - j->display_time );
521 }
522 */
523
524 if ( display_preference == Experimental ) {
525 j->reset();
526 } else {
527 reset();
528 return;
529 }
530 }
531 break;
532 case Correct:
533 /*
534 if ( j->display_time != uint64_t(-1) ) {
535 fprintf( stderr, "TIMING %ld + %ld\n", now, now - j->display_time );
536 }
537 */
538
539 if ( j->tentative_until_epoch > confirmed_epoch ) {
540 confirmed_epoch = j->tentative_until_epoch;
541
542 /*
543 fprintf( stderr, "%lc in (%d,%d) confirms epoch %lu (predicting in epoch %lu)\n",
544 j->replacement.debug_contents(), i->row_num, j->col,
545 confirmed_epoch, prediction_epoch );
546 */
547
548 }
549
550 /* When predictions come in quickly, slowly take away the glitch trigger. */
551 if ( now - j->prediction_time < GLITCH_THRESHOLD
552 && ( glitch_trigger > 0 && now - GLITCH_REPAIR_MININTERVAL >= last_quick_confirmation ) ) {
553 glitch_trigger--;
554 last_quick_confirmation = now;
555 }
556
557 /* match rest of row to the actual renditions */
558 {
559 const Renditions &actual_renditions = fb.get_cell( i->row_num, j->col )->get_renditions();
560 for ( overlay_cells_type::iterator k = j;
561 k != i->overlay_cells.end();
562 k++ ) {
563 k->replacement.get_renditions() = actual_renditions;
564 }
565 }
566
567 /* fallthrough */
568 case CorrectNoCredit:
569 j->reset();
570
571 break;
572 case Pending:
573 /* When a prediction takes a long time to be confirmed, we
574 activate the predictions even if SRTT is low */
575 if ( (now - j->prediction_time) >= GLITCH_FLAG_THRESHOLD ) {
576 glitch_trigger = GLITCH_REPAIR_COUNT * 2; /* display and underline */
577 } else if ( ((now - j->prediction_time) >= GLITCH_THRESHOLD)
578 && (glitch_trigger < GLITCH_REPAIR_COUNT) ) {
579 glitch_trigger = GLITCH_REPAIR_COUNT; /* just display */
580 }
581
582 break;
583 default:
584 break;
585 }
586 }
587
588 i = inext;
589 }
590
591 /* go through cursor predictions */
592 if ( !cursors.empty()
593 && cursor().get_validity( fb,
594 local_frame_acked, local_frame_late_acked ) == IncorrectOrExpired ) {
595 /*
596 fprintf( stderr, "Sadly, we're predicting (%d,%d) vs. (%d,%d) [tau: %ld, expiration_time=%ld, now=%ld]\n",
597 cursor().row, cursor().col,
598 fb.ds.get_cursor_row(),
599 fb.ds.get_cursor_col(),
600 cursor().tentative_until_epoch,
601 cursor().expiration_time,
602 now );
603 */
604 if ( display_preference == Experimental ) {
605 cursors.clear();
606 } else {
607 reset();
608 return;
609 }
610 }
611
612 /* NB: switching from list to another STL container could break this code.
613 So we don't use the cursors_type typedef. */
614 for ( list<ConditionalCursorMove>::iterator it = cursors.begin();
615 it != cursors.end(); ) {
616 if ( it->get_validity( fb, local_frame_acked, local_frame_late_acked ) != Pending ) {
617 it = cursors.erase( it );
618 } else {
619 it++;
620 }
621 }
622}

624ConditionalOverlayRow & PredictionEngine::get_or_make_row( int row_num, int num_cols )
625{
626 overlays_type::iterator it =
627 find_if( overlays.begin(), overlays.end(),
628 bind2nd( mem_fun_ref( &ConditionalOverlayRow::row_num_eq ), row_num ) );
629
630 if ( it != overlays.end() ) {
631 return *it;
632 }
633 /* make row */
634 ConditionalOverlayRow r( row_num );
635 r.overlay_cells.reserve( num_cols );
636 for ( int i = 0; i < num_cols; i++ ) {
637 r.overlay_cells.push_back( ConditionalOverlayCell( 0, i, prediction_epoch ) );
638 assert( r.overlay_cells[ i ].col == i );
639 }
640 overlays.push_back( r );
641 return overlays.back();
642}

644void PredictionEngine::new_user_byte( char the_byte, const Framebuffer &fb )
645{
646 if ( display_preference == Never ) {
647 return;
648 }
649 if ( display_preference == Experimental ) {
650 prediction_epoch = confirmed_epoch;
651 }
652
653 cull( fb );
654
655 uint64_t now = timestamp();
656
657 /* translate application-mode cursor control function to ANSI cursor control sequence */
658 if ( (last_byte == 0x1b)
659 && (the_byte == 'O') ) {
660 the_byte = '[';
661 }
662 last_byte = the_byte;
663
664 Parser::Actions actions;
665 parser.input( the_byte, actions );
666
667 for ( Parser::Actions::iterator it = actions.begin();
668 it != actions.end();
669 it++ ) {
670 Parser::Action& act = **it;
671
672 /*
673 fprintf( stderr, "Action: %s (%lc)\n",
674 act->name().c_str(), act->char_present ? act->ch : L'_' );
675 */
676
677 const std::type_info& type_act = typeid( act );
678 if ( type_act == typeid( Parser::Print ) ) {
679 /* make new prediction */
680
681 init_cursor( fb );
682
683 assert( act.char_present );
684
685 wchar_t ch = act.ch;
686 /* XXX handle wide characters */
687
688 if ( ch == 0x7f ) { /* backspace */
689 // fprintf( stderr, "Backspace.\n" );
690 ConditionalOverlayRow &the_row = get_or_make_row( cursor().row, fb.ds.get_width() );
691
692 if ( cursor().col > 0 ) {
693 cursor().col--;
694 cursor().expire( local_frame_sent + 1, now );
695
696 if ( predict_overwrite ) {
697 ConditionalOverlayCell &cell = the_row.overlay_cells[ cursor().col ];
698 cell.reset_with_orig();
699 cell.active = true;
700 cell.tentative_until_epoch = prediction_epoch;
701 cell.expire( local_frame_sent + 1, now );
702 const Cell orig_cell = *fb.get_cell();
703 cell.original_contents.push_back( orig_cell );
704 cell.replacement = orig_cell;
705 cell.replacement.clear();
706 cell.replacement.append(' ');
707 } else {
708 for ( int i = cursor().col; i < fb.ds.get_width(); i++ ) {
709 ConditionalOverlayCell &cell = the_row.overlay_cells[ i ];
710
711 cell.reset_with_orig();
712 cell.active = true;
713 cell.tentative_until_epoch = prediction_epoch;
714 cell.expire( local_frame_sent + 1, now );
715 cell.original_contents.push_back( *fb.get_cell( cursor().row, i ) );
716
717 if ( i + 2 < fb.ds.get_width() ) {
718 ConditionalOverlayCell &next_cell = the_row.overlay_cells[ i + 1 ];
719 const Cell *next_cell_actual = fb.get_cell( cursor().row, i + 1 );
720
721 if ( next_cell.active ) {
722 if ( next_cell.unknown ) {
723 cell.unknown = true;
724 } else {
725 cell.unknown = false;
726 cell.replacement = next_cell.replacement;
727 }
728 } else {
729 cell.unknown = false;
730 cell.replacement = *next_cell_actual;
731 }
732 } else {
733 cell.unknown = true;
734 }
735 }
736 }
737 }
738 } else if ( (ch < 0x20) || (wcwidth( ch ) != 1) ) {
739 /* unknown print */
740 become_tentative();
741 // fprintf( stderr, "Unknown print 0x%x\n", ch );
742 } else {
743 assert( cursor().row >= 0 );
744 assert( cursor().col >= 0 );
745 assert( cursor().row < fb.ds.get_height() );
746 assert( cursor().col < fb.ds.get_width() );
747
748 ConditionalOverlayRow &the_row = get_or_make_row( cursor().row, fb.ds.get_width() );
749
750 if ( cursor().col + 1 >= fb.ds.get_width() ) {
751 /* prediction in the last column is tricky */
752 /* e.g., emacs will show wrap character, shell will just put the character there */
753 become_tentative();
754 }
755
756 /* do the insert */
757 int rightmost_column = predict_overwrite ? cursor().col : fb.ds.get_width() - 1;
758 for ( int i = rightmost_column; i > cursor().col; i-- ) {
759 ConditionalOverlayCell &cell = the_row.overlay_cells[ i ];
760 cell.reset_with_orig();
761 cell.active = true;
762 cell.tentative_until_epoch = prediction_epoch;
763 cell.expire( local_frame_sent + 1, now );
764 cell.original_contents.push_back( *fb.get_cell( cursor().row, i ) );
765
766 ConditionalOverlayCell &prev_cell = the_row.overlay_cells[ i - 1 ];
767 const Cell *prev_cell_actual = fb.get_cell( cursor().row, i - 1 );
768
769 if ( i == fb.ds.get_width() - 1 ) {
770 cell.unknown = true;
771 } else if ( prev_cell.active ) {
772 if ( prev_cell.unknown ) {
773 cell.unknown = true;
774 } else {
775 cell.unknown = false;
776 cell.replacement = prev_cell.replacement;
777 }
778 } else {
779 cell.unknown = false;
780 cell.replacement = *prev_cell_actual;
781 }
782 }
783
784 ConditionalOverlayCell &cell = the_row.overlay_cells[ cursor().col ];
785 cell.reset_with_orig();
786 cell.active = true;
787 cell.tentative_until_epoch = prediction_epoch;
788 cell.expire( local_frame_sent + 1, now );
789 cell.replacement.get_renditions() = fb.ds.get_renditions();
790
791 /* heuristic: match renditions of character to the left */
792 if ( cursor().col > 0 ) {
793 ConditionalOverlayCell &prev_cell = the_row.overlay_cells[ cursor().col - 1 ];
794 const Cell *prev_cell_actual = fb.get_cell( cursor().row, cursor().col - 1 );
795 if ( prev_cell.active && (!prev_cell.unknown) ) {
796 cell.replacement.get_renditions() = prev_cell.replacement.get_renditions();
797 } else {
798 cell.replacement.get_renditions() = prev_cell_actual->get_renditions();
799 }
800 }
801
802 cell.replacement.clear();
803 cell.replacement.append( ch );
804 cell.original_contents.push_back( *fb.get_cell( cursor().row, cursor().col ) );
805
806 /*
807 fprintf( stderr, "[%d=>%d] Predicting %lc in row %d, col %d [tue: %lu]\n",
808 (int)local_frame_acked, (int)cell.expiration_frame,
809 ch, cursor().row, cursor().col,
810 cell.tentative_until_epoch );
811 */
812
813 cursor().expire( local_frame_sent + 1, now );
814
815 /* do we need to wrap? */
816 if ( cursor().col < fb.ds.get_width() - 1 ) {
817 cursor().col++;
818 } else {
819 become_tentative();
820 newline_carriage_return( fb );
821 }
822 }
823 } else if ( type_act == typeid( Parser::Execute ) ) {
824 if ( act.char_present && (act.ch == 0x0d) /* CR */ ) {
825 become_tentative();
826 newline_carriage_return( fb );
827 } else {
828 // fprintf( stderr, "Execute 0x%x\n", act.ch );
829 become_tentative();
830 }
831 } else if ( type_act == typeid( Parser::Esc_Dispatch ) ) {
832 // fprintf( stderr, "Escape sequence\n" );
833 become_tentative();
834 } else if ( type_act == typeid( Parser::CSI_Dispatch ) ) {
835 if ( act.char_present && (act.ch == L'C') ) { /* right arrow */
836 init_cursor( fb );
837 if ( cursor().col < fb.ds.get_width() - 1 ) {
838 cursor().col++;
839 cursor().expire( local_frame_sent + 1, now );
840 }
841 } else if ( act.char_present && (act.ch == L'D') ) { /* left arrow */
842 init_cursor( fb );
843
844 if ( cursor().col > 0 ) {
845 cursor().col--;
846 cursor().expire( local_frame_sent + 1, now );
847 }
848 } else {
849 // fprintf( stderr, "CSI sequence %lc\n", act.ch );
850 become_tentative();
851 }
852 }
853 }
854}

856void PredictionEngine::newline_carriage_return( const Framebuffer &fb )
857{
858 uint64_t now = timestamp();
859 init_cursor( fb );
860 cursor().col = 0;
861 if ( cursor().row == fb.ds.get_height() - 1 ) {
862 /* Don't try to predict scroll until we have versioned cell predictions */
863 /*
864 for ( overlays_type::iterator i = overlays.begin();
865 i != overlays.end();
866 i++ ) {
867 i->row_num--;
868 for ( overlay_cells_type::iterator j = i->overlay_cells.begin();
869 j != i->overlay_cells.end();
870 j++ ) {
871 if ( j->active ) {
872 j->expire( local_frame_sent + 1, now );
873 }
874 }
875 }
876 */
877
878 /* make blank prediction for last row */
879 ConditionalOverlayRow &the_row = get_or_make_row( cursor().row, fb.ds.get_width() );
880 for ( overlay_cells_type::iterator j = the_row.overlay_cells.begin();
881 j != the_row.overlay_cells.end();
882 j++ ) {
883 j->active = true;
884 j->tentative_until_epoch = prediction_epoch;
885 j->expire( local_frame_sent + 1, now );
886 j->replacement.clear();
887 }
888 } else {
889 cursor().row++;
890 }

904
905bool PredictionEngine::active( void ) const
906{
907 if ( !cursors.empty() ) {
908 return true;
909 }
910
911 for ( overlays_type::const_iterator i = overlays.begin();
912 i != overlays.end();
913 i++ ) {
914 for ( overlay_cells_type::const_iterator j = i->overlay_cells.begin();
915 j != i->overlay_cells.end();
916 j++ ) {
917 if ( j->active ) {
918 return true;
919 }
920 }
921 }
922
923 return false;
924}
