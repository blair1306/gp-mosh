128/* Simple spinloop */
129static void spin( void )
130{
131 static unsigned int spincount = 0;
132 spincount++;
133
134 if ( spincount > 10 ) {
135 struct timespec req;
136 req.tv_sec = 0;
137 req.tv_nsec = 100000000; /* 0.1 sec */
138 nanosleep( &req, NULL );
139 freeze_timestamp();
140 }
141}

143static string get_SSH_IP( void )
144{
145 const char *SSH_CONNECTION = getenv( "SSH_CONNECTION" );
146 if ( !SSH_CONNECTION ) { /* Older sshds don't set this */
147 fputs( "Warning: SSH_CONNECTION not found; binding to any interface.\n", stderr );
148 return string( "" );
149 }
150 istringstream ss( SSH_CONNECTION );
151 string dummy, local_interface_IP;
152 ss >> dummy >> dummy >> local_interface_IP;
153 if ( !ss ) {
154 fputs( "Warning: Could not parse SSH_CONNECTION; binding to any interface.\n", stderr );
155 return string( "" );
156 }
157
158 /* Strip IPv6 prefix. */
159 const char IPv6_prefix[] = "::ffff:";
160
161 if ( ( local_interface_IP.length() > strlen( IPv6_prefix ) )
162 && ( 0 == strncasecmp( local_interface_IP.c_str(), IPv6_prefix, strlen( IPv6_prefix ) ) ) ) {
163 return local_interface_IP.substr( strlen( IPv6_prefix ) );
164 }
165
166 return local_interface_IP;
167}

169int main( int argc, char *argv[] )
170{
171 /* For security, make sure we don't dump core */
172 Crypto::disable_dumping_core();
173
174 /* Detect edge case */
175 fatal_assert( argc > 0 );
176
177 const char *desired_ip = NULL;
178 string desired_ip_str;
179 const char *desired_port = NULL;
180 string command_path;
181 char **command_argv = NULL;
182 int colors = 0;
183 unsigned int verbose = 0; /* don't close stdin/stdout/stderr */
184 /* Will cause mosh-server not to correctly detach on old versions of sshd. */
185 list<string> locale_vars;
186
187 /* strip off command */
188 for ( int i = 1; i < argc; i++ ) {
189 if ( 0 == strcmp( argv[ i ], "--help" ) || 0 == strcmp( argv[ i ], "-h" ) ) {
190 print_usage( stdout, argv[ 0 ] );
191 exit( 0 );
192 }
193 if ( 0 == strcmp( argv[ i ], "--version" ) ) {
194 print_version( stdout );
195 exit( 0 );
196 }
197 if ( 0 == strcmp( argv[ i ], "--" ) ) { /* -- is mandatory */
198 if ( i != argc - 1 ) {
199 command_argv = argv + i + 1;
200 }
201 argc = i; /* rest of options before -- */
202 break;
203 }
204 }
205
206 /* Parse new command-line syntax */
207 if ( (argc >= 2)
208 && (strcmp( argv[ 1 ], "new" ) == 0) ) {
209 /* new option syntax */
210 int opt;
211 while ( (opt = getopt( argc - 1, argv + 1, "@:i:p:c:svl:" )) != -1 ) {
212 switch ( opt ) {
213 /*
214 * This undocumented option does nothing but eat its argument.
215 * Useful in scripting where you prepend something to a
216 * mosh-server argv, and might end up with something like
217 * "mosh-server new -v new -c 256", now you can say
218 * "mosh-server new -v -@ new -c 256" to discard the second
219 * "new".
220 */
221 case '@':
222 break;
223 case 'i':
224 desired_ip = optarg;
225 break;
226 case 'p':
227 desired_port = optarg;
228 break;
229 case 's':
230 desired_ip = NULL;
231 desired_ip_str = get_SSH_IP();
232 if ( !desired_ip_str.empty() ) {
233 desired_ip = desired_ip_str.c_str();
234 fatal_assert( desired_ip );
235 }
236 break;
237 case 'c':
238 try {
239 colors = myatoi( optarg );
240 } catch ( const CryptoException & ) {
241 fprintf( stderr, "%s: Bad number of colors (%s)\n", argv[ 0 ], optarg );
242 print_usage( stderr, argv[ 0 ] );
243 exit( 1 );
244 }
245 break;
246 case 'v':
247 verbose++;
248 break;
249 case 'l':
250 locale_vars.push_back( string( optarg ) );
251 break;
252 default:
253 /* don't die on unknown options */
254 print_usage( stderr, argv[ 0 ] );
255 break;
256 }
257 }
258 } else if ( argc == 1 ) {
259 /* legacy argument parsing for older client wrapper script */
260 /* do nothing */
261 } else if ( argc == 2 ) {
262 desired_ip = argv[ 1 ];
263 } else if ( argc == 3 ) {
264 desired_ip = argv[ 1 ];
265 desired_port = argv[ 2 ];
266 } else {
267 print_usage( stderr, argv[ 0 ] );
268 exit( 1 );
269 }
270
271 /* Sanity-check arguments */
272 int dpl, dph;
273 if ( desired_port && ! Connection::parse_portrange( desired_port, dpl, dph ) ) {
274 fprintf( stderr, "%s: Bad UDP port range (%s)\n", argv[ 0 ], desired_port );
275 print_usage( stderr, argv[ 0 ] );
276 exit( 1 );
277 }
278
279 bool with_motd = false;
280
281 /* Get shell */
282 char *my_argv[ 2 ];
283 string shell_name;
284 if ( !command_argv ) {
285 /* get shell name */
286 const char *shell = getenv( "SHELL" );
287 if ( shell == NULL ) {
288 struct passwd *pw = getpwuid( getuid() );
289 if ( pw == NULL ) {
290 perror( "getpwuid" );
291 exit( 1 );
292 }
293 shell = pw->pw_shell;
294 }
295
296 string shell_path( shell );
297 if ( shell_path.empty() ) { /* empty shell means Bourne shell */
298 shell_path = _PATH_BSHELL;
299 }
300
301 command_path = shell_path;
302
303 size_t shell_slash( shell_path.rfind('/') );
304 if ( shell_slash == string::npos ) {
305 shell_name = shell_path;
306 } else {
307 shell_name = shell_path.substr(shell_slash + 1);
308 }
309
310 /* prepend '-' to make login shell */
311 shell_name = '-' + shell_name;
312
313 my_argv[ 0 ] = const_cast<char *>( shell_name.c_str() );
314 my_argv[ 1 ] = NULL;
315 command_argv = my_argv;
316
317 with_motd = true;
318 }
319
320 if ( command_path.empty() ) {
321 command_path = command_argv[0];
322 }
323
324 /* Adopt implementation locale */
325 set_native_locale();
326 if ( !is_utf8_locale() ) {
327 /* save details for diagnostic */
328 LocaleVar native_ctype = get_ctype();
329 string native_charset( locale_charset() );
330
331 /* apply locale-related environment variables from client */
332 clear_locale_variables();
333 for ( list<string>::const_iterator i = locale_vars.begin();
334 i != locale_vars.end();
335 i++ ) {
336 char *env_string = strdup( i->c_str() );
337 fatal_assert( env_string );
338 if ( 0 != putenv( env_string ) ) {
339 perror( "putenv" );
340 }
341 }
342
343 /* check again */
344 set_native_locale();
345 if ( !is_utf8_locale() ) {
346 LocaleVar client_ctype = get_ctype();
347 string client_charset( locale_charset() );
348
349 fprintf( stderr, "mosh-server needs a UTF-8 native locale to run.\n\n"
350 "Unfortunately, the local environment (%s) specifies\n"
351 "the character set \"%s\",\n\n"
352 "The client-supplied environment (%s) specifies\n"
353 "the character set \"%s\".\n\n",
354 native_ctype.str().c_str(), native_charset.c_str(), client_ctype.str().c_str(), client_charset.c_str() );
355 int unused __attribute((unused)) = system( "locale" );
356 exit( 1 );
357 }
358 }
359
360 try {
361 return run_server( desired_ip, desired_port, command_path, command_argv, colors, verbose, with_motd );
362 } catch ( const Network::NetworkException &e ) {
363 fprintf( stderr, "Network exception: %s\n",
364 e.what() );
365 return 1;
366 } catch ( const Crypto::CryptoException &e ) {
367 fprintf( stderr, "Crypto exception: %s\n",
368 e.what() );
369 return 1;
370 }
371}

373static int run_server( const char *desired_ip, const char *desired_port,
374 const string &command_path, char *command_argv[],
375 const int colors, unsigned int verbose, bool with_motd ) {
376 /* get network idle timeout */
377 long network_timeout = 0;
378 char *timeout_envar = getenv( "MOSH_SERVER_NETWORK_TMOUT" );
379 if ( timeout_envar && *timeout_envar ) {
380 errno = 0;
381 char *endptr;
382 network_timeout = strtol( timeout_envar, &endptr, 10 );
383 if ( *endptr != '\0' || ( network_timeout == 0 && errno == EINVAL ) ) {
384 fputs( "MOSH_SERVER_NETWORK_TMOUT not a valid integer, ignoring\n", stderr );
385 } else if ( network_timeout < 0 ) {
386 fputs( "MOSH_SERVER_NETWORK_TMOUT is negative, ignoring\n", stderr );
387 network_timeout = 0;
388 }
389 }
390 /* get network signaled idle timeout */
391 long network_signaled_timeout = 0;
392 char *signal_envar = getenv( "MOSH_SERVER_SIGNAL_TMOUT" );
393 if ( signal_envar && *signal_envar ) {
394 errno = 0;
395 char *endptr;
396 network_signaled_timeout = strtol( signal_envar, &endptr, 10 );
397 if ( *endptr != '\0' || ( network_signaled_timeout == 0 && errno == EINVAL ) ) {
398 fputs( "MOSH_SERVER_SIGNAL_TMOUT not a valid integer, ignoring\n", stderr );
399 } else if ( network_signaled_timeout < 0 ) {
400 fputs( "MOSH_SERVER_SIGNAL_TMOUT is negative, ignoring\n", stderr );
401 network_signaled_timeout = 0;
402 }
403 }
404 /* get initial window size */
405 struct winsize window_size;
406 if ( ioctl( STDIN_FILENO, TIOCGWINSZ, &window_size ) < 0 ||
407 window_size.ws_col == 0 ||
408 window_size.ws_row == 0 ) {
409 /* Fill in sensible defaults. */
410 /* They will be overwritten by client on first connection. */
411 memset( &window_size, 0, sizeof( window_size ) );
412 window_size.ws_col = 80;
413 window_size.ws_row = 24;
414 }
415
416 /* open parser and terminal */
417 Terminal::Complete terminal( window_size.ws_col, window_size.ws_row );
418
419 /* open network */
420 Network::UserStream blank;
421 typedef shared::shared_ptr<ServerConnection> NetworkPointer;
422 NetworkPointer network( new ServerConnection( terminal, blank, desired_ip, desired_port ) );
423
424 network->set_verbose( verbose );
425 Select::set_verbose( verbose );
426
427 /*
428 * If mosh-server is run on a pty, then typeahead may echo and break mosh.pl's
429 * detection of the MOSH CONNECT message. Print it on a new line to bodge
430 * around that.
431 */
432 if ( isatty( STDIN_FILENO ) ) {
433 puts( "\r\n" );
434 }
435 printf( "MOSH CONNECT %s %s\n", network->port().c_str(), network->get_key().c_str() );
436
437 /* don't let signals kill us */
438 struct sigaction sa;
439 sa.sa_handler = SIG_IGN;
440 sa.sa_flags = 0;
441 fatal_assert( 0 == sigfillset( &sa.sa_mask ) );
442 fatal_assert( 0 == sigaction( SIGHUP, &sa, NULL ) );
443 fatal_assert( 0 == sigaction( SIGPIPE, &sa, NULL ) );
444
445
446 /* detach from terminal */
447 fflush( stdout );
448 fflush( stderr );
449 pid_t the_pid = fork();
450 if ( the_pid < 0 ) {
451 perror( "fork" );
452 } else if ( the_pid > 0 ) {
453 fputs( "\nmosh-server (" PACKAGE_STRING ") [build " BUILD_VERSION "]\n"
454 "Copyright 2012 Keith Winstein <mosh-devel@mit.edu>\n"
455 "License GPLv3+: GNU GPL version 3 or later <http://gnu.org/licenses/gpl.html>.\n"
456 "This is free software: you are free to change and redistribute it.\n"
457 "There is NO WARRANTY, to the extent permitted by law.\n\n", stderr );
458
459 fprintf( stderr, "[mosh-server detached, pid = %d]\n", static_cast<int>(the_pid) );
460#ifndef HAVE_IUTF8
461 fputs( "\nWarning: termios IUTF8 flag not defined.\n"
462 "Character-erase of multibyte character sequence\n"
463 "probably does not work properly on this platform.\n", stderr );
464#endif /* HAVE_IUTF8 */
465
466 fflush( stdout );
467 fflush( stderr );
468 if ( isatty( STDOUT_FILENO ) ) {
469 tcdrain( STDOUT_FILENO );
470 }
471 if ( isatty( STDERR_FILENO ) ) {
472 tcdrain( STDERR_FILENO );
473 }
474 _exit( 0 );
475 }
476
477 int master;
478
479 /* close file descriptors */
480 if ( verbose == 0 ) {
481 /* Necessary to properly detach on old versions of sshd (e.g. RHEL/CentOS 5.0). */
482 int nullfd;
483
484 nullfd = open( "/dev/null", O_RDWR );
485 if ( nullfd == -1 ) {
486 perror( "open" );
487 exit( 1 );
488 }
489
490 if ( dup2 ( nullfd, STDIN_FILENO ) < 0 ||
491 dup2 ( nullfd, STDOUT_FILENO ) < 0 ||
492 dup2 ( nullfd, STDERR_FILENO ) < 0 ) {
493 perror( "dup2" );
494 exit( 1 );
495 }
496
497 if ( close( nullfd ) < 0 ) {
498 perror( "close" );
499 exit( 1 );
500 }
501 }
502
503 char utmp_entry[ 64 ] = { 0 };
504 snprintf( utmp_entry, 64, "mosh [%ld]", static_cast<long int>( getpid() ) );
505
506 /* Fork child process */
507 pid_t child = forkpty( &master, NULL, NULL, &window_size );
508
509 if ( child == -1 ) {
510 perror( "forkpty" );
511 exit( 1 );
512 }
513
514 if ( child == 0 ) {
515 /* child */
516
517 /* reenable signals */
518 struct sigaction sa;
519 sa.sa_handler = SIG_DFL;
520 sa.sa_flags = 0;
521 fatal_assert( 0 == sigfillset( &sa.sa_mask ) );
522 fatal_assert( 0 == sigaction( SIGHUP, &sa, NULL ) );
523 fatal_assert( 0 == sigaction( SIGPIPE, &sa, NULL ) );
524
525 /* close server-related file descriptors */
526 network.reset();
527
528 /* set IUTF8 if available */
529#ifdef HAVE_IUTF8
530 struct termios child_termios;
531 if ( tcgetattr( STDIN_FILENO, &child_termios ) < 0 ) {
532 perror( "tcgetattr" );
533 exit( 1 );
534 }
535
536 child_termios.c_iflag |= IUTF8;
537
538 if ( tcsetattr( STDIN_FILENO, TCSANOW, &child_termios ) < 0 ) {
539 perror( "tcsetattr" );
540 exit( 1 );
541 }
542#endif /* HAVE_IUTF8 */
543
544 /* set TERM */
545 const char default_term[] = "xterm";
546 const char color_term[] = "xterm-256color";
547
548 if ( setenv( "TERM", (colors == 256) ? color_term : default_term, true ) < 0 ) {
549 perror( "setenv" );
550 exit( 1 );
551 }
552
553 /* ask ncurses to send UTF-8 instead of ISO 2022 for line-drawing chars */
554 if ( setenv( "NCURSES_NO_UTF8_ACS", "1", true ) < 0 ) {
555 perror( "setenv" );
556 exit( 1 );
557 }
558
559 /* clear STY environment variable so GNU screen regards us as top level */
560 if ( unsetenv( "STY" ) < 0 ) {
561 perror( "unsetenv" );
562 exit( 1 );
563 }
564
565 chdir_homedir();
566
567 if ( with_motd && (!motd_hushed()) ) {
568 // On illumos motd is printed by /etc/profile.
569#ifndef __sun
570 // For Ubuntu, try and print one of {,/var}/run/motd.dynamic.
571 // This file is only updated when pam_motd is run, but when
572 // mosh-server is run in the usual way with ssh via the script,
573 // this always happens.
574 // XXX Hackish knowledge of Ubuntu PAM configuration.
575 // But this seems less awful than build-time detection with autoconf.
576 if (!print_motd("/run/motd.dynamic")) {
577 print_motd("/var/run/motd.dynamic");
578 }
579 // Always print traditional /etc/motd.
580 print_motd("/etc/motd");
581#endif
582 warn_unattached( utmp_entry );
583 }
584
585 /* Wait for parent to release us. */
586 char linebuf[81];
587 if (fgets(linebuf, sizeof linebuf, stdin) == NULL) {
588 perror( "parent signal" );
589 _exit( 1 );
590 }
591
592 Crypto::reenable_dumping_core();
593
594 if ( execvp( command_path.c_str(), command_argv ) < 0 ) {
595 perror( "execvp" );
596 _exit( 1 );
597 }
598 } else {
599 /* parent */
600
601 /* Drop unnecessary privileges */
602#ifdef HAVE_PLEDGE
603 /* OpenBSD pledge() syscall */
604 if ( pledge( "stdio inet tty", NULL )) {
605 perror( "pledge() failed" );
606 exit( 1 );
607 }
608#endif
609
610#ifdef HAVE_UTEMPTER
611 /* make utmp entry */
612 utempter_add_record( master, utmp_entry );
613#endif
614
615 try {
616 serve( master, terminal, *network, network_timeout, network_signaled_timeout );
617 } catch ( const Network::NetworkException &e ) {
618 fprintf( stderr, "Network exception: %s\n",
619 e.what() );
620 } catch ( const Crypto::CryptoException &e ) {
621 fprintf( stderr, "Crypto exception: %s\n",
622 e.what() );
623 }
624
625 #ifdef HAVE_UTEMPTER
626 utempter_remove_record( master );
627 #endif
628
629 if ( close( master ) < 0 ) {
630 perror( "close" );
631 exit( 1 );
632 }
633 }
634
635 fputs( "\n[mosh-server is exiting.]\n", stdout );
636
637 return 0;
638}

640static void serve( int host_fd, Terminal::Complete &terminal, ServerConnection &network, long network_timeout, long network_signaled_timeout )
641{
642 /* scale timeouts */
643 const uint64_t network_timeout_ms = static_cast<uint64_t>( network_timeout ) * 1000;
644 const uint64_t network_signaled_timeout_ms = static_cast<uint64_t>( network_signaled_timeout ) * 1000;
645 /* prepare to poll for events */
646 Select &sel = Select::get_instance();
647 sel.add_signal( SIGTERM );
648 sel.add_signal( SIGINT );
649 sel.add_signal( SIGUSR1 );
650
651 uint64_t last_remote_num = network.get_remote_state_num();
652
653 #ifdef HAVE_UTEMPTER
654 bool connected_utmp = false;
655
656 Addr saved_addr;
657 socklen_t saved_addr_len = 0;
658 #endif
659
660 bool child_released = false;
661
662 while ( true ) {
663 try {
664 static const uint64_t timeout_if_no_client = 60000;
665 int timeout = INT_MAX;
666 uint64_t now = Network::timestamp();
667
668 timeout = min( timeout, network.wait_time() );
669 timeout = min( timeout, terminal.wait_time( now ) );
670 if ( (!network.get_remote_state_num())
671 || network.shutdown_in_progress() ) {
672 timeout = min( timeout, 5000 );
673 }
674 /*
675 * The server goes completely asleep if it has no remote peer.
676 * We may want to wake up sooner.
677 */
678 if ( network_timeout_ms ) {
679 int64_t network_sleep = network_timeout_ms -
680 ( now - network.get_latest_remote_state().timestamp );
681 if ( network_sleep < 0 ) {
682 network_sleep = 0;
683 } else if ( network_sleep > INT_MAX ) {
684 /* 24 days might be too soon. That's OK. */
685 network_sleep = INT_MAX;
686 }
687 timeout = min( timeout, static_cast<int>(network_sleep) );
688 }
689
690 /* poll for events */
691 sel.clear_fds();
692 std::vector< int > fd_list( network.fds() );
693 assert( fd_list.size() == 1 ); /* servers don't hop */
694 int network_fd = fd_list.back();
695 sel.add_fd( network_fd );
696 if ( !network.shutdown_in_progress() ) {
697 sel.add_fd( host_fd );
698 }
699
700 int active_fds = sel.select( timeout );
701 if ( active_fds < 0 ) {
702 perror( "select" );
703 break;
704 }
705
706 now = Network::timestamp();
707 uint64_t time_since_remote_state = now - network.get_latest_remote_state().timestamp;
708 string terminal_to_host;
709
710 if ( sel.read( network_fd ) ) {
711 /* packet received from the network */
712 network.recv();
713
714 /* is new user input available for the terminal? */
715 if ( network.get_remote_state_num() != last_remote_num ) {
716 last_remote_num = network.get_remote_state_num();
717
718
719 Network::UserStream us;
720 us.apply_string( network.get_remote_diff() );
721 /* apply userstream to terminal */
722 for ( size_t i = 0; i < us.size(); i++ ) {
723 const Parser::Action &action = us.get_action( i );
724 if ( typeid( action ) == typeid( Parser::Resize ) ) {
725 /* apply only the last consecutive Resize action */
726 while ( i < us.size() - 1 ) {
727 const Parser::Action &next = us.get_action( i + 1 );
728 if ( typeid( next ) != typeid( Parser::Resize ) ) {
729 break;
730 }
731 i++;
732 }
733 /* tell child process of resize */
734 const Parser::Resize &res = static_cast<const Parser::Resize &>( action );
735 struct winsize window_size;
736 if ( ioctl( host_fd, TIOCGWINSZ, &window_size ) < 0 ) {
737 perror( "ioctl TIOCGWINSZ" );
738 network.start_shutdown();
739 }
740 window_size.ws_col = res.width;
741 window_size.ws_row = res.height;
742 if ( ioctl( host_fd, TIOCSWINSZ, &window_size ) < 0 ) {
743 perror( "ioctl TIOCSWINSZ" );
744 network.start_shutdown();
745 }
746 }
747 terminal_to_host += terminal.act( action );
748 }
749
750 if ( !us.empty() ) {
751 /* register input frame number for future echo ack */
752 terminal.register_input_frame( last_remote_num, now );
753 }
754
755 /* update client with new state of terminal */
756 if ( !network.shutdown_in_progress() ) {
757 network.set_current_state( terminal );
758 }
759
760 #ifdef HAVE_UTEMPTER
761 /* update utmp entry if we have become "connected" */
762 if ( (!connected_utmp)
763 || saved_addr_len != network.get_remote_addr_len()
764 || memcmp( &saved_addr, &network.get_remote_addr(),
765 saved_addr_len ) != 0 ) {
766 utempter_remove_record( host_fd );
767
768 saved_addr = network.get_remote_addr();
769 saved_addr_len = network.get_remote_addr_len();
770
771 char host[ NI_MAXHOST ];
772 int errcode = getnameinfo( &saved_addr.sa, saved_addr_len,
773 host, sizeof( host ), NULL, 0,
774 NI_NUMERICHOST );
775 if ( errcode != 0 ) {
776 throw NetworkException( std::string( "serve: getnameinfo: " ) + gai_strerror( errcode ), 0 );
777 }
778
779 char tmp[ 64 ];
780 snprintf( tmp, 64, "%s via mosh [%d]", host, getpid() );
781 utempter_add_record( host_fd, tmp );
782
783 connected_utmp = true;
784 }
785 #endif
786
787 /* Tell child to start login session. */
788 if ( !child_released ) {
789 if ( swrite( host_fd, "\n", 1 ) < 0) {
790 perror( "child release" );
791 _exit( 1 );
792 }
793 child_released = true;
794 }
795 }
796 }
797
798 if ( (!network.shutdown_in_progress()) && sel.read( host_fd ) ) {
799 /* input from the host needs to be fed to the terminal */
800 const int buf_size = 16384;
801 char buf[ buf_size ];
802
803 /* fill buffer if possible */
804 ssize_t bytes_read = read( host_fd, buf, buf_size );
805
806 /* If the pty slave is closed, reading from the master can fail with
807 EIO (see #264). So we treat errors on read() like EOF. */
808 if ( bytes_read <= 0 ) {
809 network.start_shutdown();
810 } else {
811 terminal_to_host += terminal.act( string( buf, bytes_read ) );
812
813 /* update client with new state of terminal */
814 network.set_current_state( terminal );
815 }
816 }
817
818 /* write user input and terminal writeback to the host */
819 if ( swrite( host_fd, terminal_to_host.c_str(), terminal_to_host.length() ) < 0 ) {
820 network.start_shutdown();
821 }
822
823 bool idle_shutdown = false;
824 if ( network_timeout_ms &&
825 network_timeout_ms <= time_since_remote_state ) {
826 idle_shutdown = true;
827 fprintf( stderr, "Network idle for %llu seconds.\n",
828 static_cast<unsigned long long>( time_since_remote_state / 1000 ) );
829 }
830 if ( sel.signal( SIGUSR1 )
831 && ( !network_signaled_timeout_ms || network_signaled_timeout_ms <= time_since_remote_state ) ) {
832 idle_shutdown = true;
833 fprintf( stderr, "Network idle for %llu seconds when SIGUSR1 received\n",
834 static_cast<unsigned long long>( time_since_remote_state / 1000 ) );
835 }
836
837 if ( sel.any_signal() || idle_shutdown ) {
838 /* shutdown signal */
839 if ( network.has_remote_addr() && (!network.shutdown_in_progress()) ) {
840 network.start_shutdown();
841 } else {
842 break;
843 }
844 }
845
846 /* quit if our shutdown has been acknowledged */
847 if ( network.shutdown_in_progress() && network.shutdown_acknowledged() ) {
848 break;
849 }
850
851 /* quit after shutdown acknowledgement timeout */
852 if ( network.shutdown_in_progress() && network.shutdown_ack_timed_out() ) {
853 break;
854 }
855
856 /* quit if we received and acknowledged a shutdown request */
857 if ( network.counterparty_shutdown_ack_sent() ) {
858 break;
859 }
860
861 #ifdef HAVE_UTEMPTER
862 /* update utmp if has been more than 30 seconds since heard from client */
863 if ( connected_utmp
864 && time_since_remote_state > 30000 ) {
865 utempter_remove_record( host_fd );
866
867 char tmp[ 64 ];
868 snprintf( tmp, 64, "mosh [%d]", getpid() );
869 utempter_add_record( host_fd, tmp );
870
871 connected_utmp = false;
872 }
873 #endif
874
875 if ( terminal.set_echo_ack( now ) && !network.shutdown_in_progress() ) {
876 /* update client with new echo ack */
877 network.set_current_state( terminal );
878 }
879
880 if ( !network.get_remote_state_num()
881 && time_since_remote_state >= timeout_if_no_client ) {
882 fprintf( stderr, "No connection within %llu seconds.\n",
883 static_cast<unsigned long long>( timeout_if_no_client / 1000 ) );
884 break;
885 }
886
887 network.tick();
888 } catch ( const Network::NetworkException &e ) {
889 fprintf( stderr, "%s\n", e.what() );
890 spin();
891 } catch ( const Crypto::CryptoException &e ) {
892 if ( e.fatal ) {
893 throw;
894 } else {
895 fprintf( stderr, "Crypto exception: %s\n", e.what() );
896 }
897 }
898 }
899}

901/* Print the motd from a given file, if available */
902static bool print_motd( const char *filename )
903{
904 FILE *motd = fopen( filename, "r" );
905 if ( !motd ) {
906 return false;
907 }
908
909 const int BUFSIZE = 256;
910
911 char buffer[ BUFSIZE ];
912 while ( 1 ) {
913 size_t bytes_read = fread( buffer, 1, BUFSIZE, motd );
914 if ( bytes_read == 0 ) {
915 break; /* don't report error */
916 }
917 size_t bytes_written = fwrite( buffer, 1, bytes_read, stdout );
918 if ( bytes_written == 0 ) {
919 break;
920 }
921 }
922
923 fclose( motd );
924 return true;
925}

944 perror( "setenv" );

964static void warn_unattached( const string & ignore_entry )
965{
966#ifdef HAVE_UTMPX_H
967 /* get username */
968 const struct passwd *pw = getpwuid( getuid() );
969 if ( pw == NULL ) {
970 perror( "getpwuid" );
971 /* non-fatal */
972 return;
973 }
974
975 const string username( pw->pw_name );
976
977 /* look for unattached sessions */
978 vector< string > unattached_mosh_servers;
979
980 while ( struct utmpx *entry = getutxent() ) {
981 if ( (entry->ut_type == USER_PROCESS)
982 && (username == string( entry->ut_user )) ) {
983 /* does line show unattached mosh session */
984 string text( entry->ut_host );
985 if ( (text.size() >= 5)
986 && (text.substr( 0, 5 ) == "mosh ")
987 && (text[ text.size() - 1 ] == ']')
988 && (text != ignore_entry)
989 && device_exists( entry->ut_line ) ) {
990 unattached_mosh_servers.push_back( text );
991 }
992 }
993 }
994
995 /* print out warning if necessary */
996 if ( unattached_mosh_servers.empty() ) {
997 return;
998 } else if ( unattached_mosh_servers.size() == 1 ) {
999 printf( "\033[37;44mMosh: You have a detached Mosh session on this server (%s).\033[m\n\n",
1000 unattached_mosh_servers.front().c_str() );
1001 } else {
1002 string pid_string;
1003
1004 for ( vector< string >::const_iterator it = unattached_mosh_servers.begin();
1005 it != unattached_mosh_servers.end();
1006 it++ ) {
1007 pid_string += " - " + *it + "\n";
1008 }
1009
1010 printf( "\033[37;44mMosh: You have %d detached Mosh sessions on this server, with PIDs:\n%s\033[m\n",
1011 (int)unattached_mosh_servers.size(), pid_string.c_str() );
1012 }
1013#endif /* HAVE_UTMPX_H */
