33#include <unistd.h>
34#include <string.h>
35#include <stdio.h>
36
37#include "swrite.h"
38
39int swrite( int fd, const char *str, ssize_t len )
40{
41 ssize_t total_bytes_written = 0;
42 ssize_t bytes_to_write = ( len >= 0 ) ? len : (ssize_t) strlen( str );
43 while ( total_bytes_written < bytes_to_write ) {
44 ssize_t bytes_written = write( fd, str + total_bytes_written,
45 bytes_to_write - total_bytes_written );
46 if ( bytes_written <= 0 ) {
47 perror( "write" );
48 return -1;
49 }
50 total_bytes_written += bytes_written;
51 }
52
53 return 0;
54}
