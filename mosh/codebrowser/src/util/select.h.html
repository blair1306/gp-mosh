33#ifndef SELECT_HPP
34#define SELECT_HPP
35
36#include <string.h>
37#include <errno.h>
38#include <signal.h>
39#include <sys/select.h>
40#include <assert.h>
41
42#include "fatal_assert.h"
43#include "timestamp.h"
44
45/* Convenience wrapper for pselect(2).
46
47 Any signals blocked by calling sigprocmask() outside this code will still be
48 received during Select::select(). So don't do that. */
49
50class Select {
51public:
52 static Select &get_instance( void ) {
53 /* COFU may or may not be thread-safe, depending on compiler */
54 static Select instance;
55 return instance;
56 }
57
58private:
59 Select()
60 : max_fd( -1 )
61 /* These initializations are not used; they are just
62 here to appease -Weffc++. */
63 , all_fds( dummy_fd_set )
64 , read_fds( dummy_fd_set )
65 , empty_sigset( dummy_sigset )
66 , consecutive_polls( 0 )
67 {
68 FD_ZERO( &all_fds );
69 FD_ZERO( &read_fds );
70
71 clear_got_signal();
72 fatal_assert( 0 == sigemptyset( &empty_sigset ) );
73 }
74
75 void clear_got_signal( void )
76 {
77 memset( got_signal, 0, sizeof( got_signal ) );
78 }
79
80 /* not implemented */
81 Select( const Select & );
82 Select &operator=( const Select & );
83
84public:
85 void add_fd( int fd )
86 {
87 if ( fd > max_fd ) {
88 max_fd = fd;
89 }
90 FD_SET( fd, &all_fds );
91 }
92
93 void clear_fds( void )
94 {
95 FD_ZERO( &all_fds );
96 }
97
98 static void add_signal( int signum )
99 {
100 fatal_assert( signum >= 0 );
101 fatal_assert( signum <= MAX_SIGNAL_NUMBER );
102
103 /* Block the signal so we don't get it outside of pselect(). */
104 sigset_t to_block;
105 fatal_assert( 0 == sigemptyset( &to_block ) );
106 fatal_assert( 0 == sigaddset( &to_block, signum ) );
107 fatal_assert( 0 == sigprocmask( SIG_BLOCK, &to_block, NULL ) );
108
109 /* Register a handler, which will only be called when pselect()
110 is interrupted by a (possibly queued) signal. */
111 struct sigaction sa;
112 sa.sa_flags = 0;
113 sa.sa_handler = &handle_signal;
114 fatal_assert( 0 == sigfillset( &sa.sa_mask ) );
115 fatal_assert( 0 == sigaction( signum, &sa, NULL ) );
116 }
117
118 /* timeout unit: milliseconds; negative timeout means wait forever */
119 int select( int timeout )
120 {
121 memcpy( &read_fds, &all_fds, sizeof( read_fds ) );
122 clear_got_signal();
123
124 /* Rate-limit and warn about polls. */
125 if ( verbose > 1 && timeout == 0 ) {
126 fprintf( stderr, "%s: got poll (timeout 0)\n", __func__ );
127 }
128 if ( timeout == 0 && ++consecutive_polls >= MAX_POLLS ) {
129 if ( verbose > 1 && consecutive_polls == MAX_POLLS ) {
130 fprintf( stderr, "%s: got %d polls, rate limiting.\n", __func__, MAX_POLLS );
131 }
132 timeout = 1;
133 } else if ( timeout != 0 && consecutive_polls ) {
134 if ( verbose > 1 && consecutive_polls >= MAX_POLLS ) {
135 fprintf( stderr, "%s: got %d consecutive polls\n", __func__, consecutive_polls );
136 }
137 consecutive_polls = 0;
138 }
139
140#ifdef HAVE_PSELECT
141 struct timespec ts;
142 struct timespec *tsp = NULL;
143
144 if ( timeout >= 0 ) {
145 ts.tv_sec = timeout / 1000;
146 ts.tv_nsec = 1000000 * (long( timeout ) % 1000);
147 tsp = &ts;
148 }
149
150 int ret = ::pselect( max_fd + 1, &read_fds, NULL, NULL, tsp, &empty_sigset );
151#else
152 struct timeval tv;
153 struct timeval *tvp = NULL;
154 sigset_t old_sigset;
155
156 if ( timeout >= 0 ) {
157 tv.tv_sec = timeout / 1000;
158 tv.tv_usec = 1000 * (long( timeout ) % 1000);
159 tvp = &tv;
160 }
161
162 int ret = sigprocmask( SIG_SETMASK, &empty_sigset, &old_sigset );
163 if ( ret != -1 ) {
164 ret = ::select( max_fd + 1, &read_fds, NULL, NULL, tvp );
165 sigprocmask( SIG_SETMASK, &old_sigset, NULL );
166 }
167#endif
168
169 if ( ret == 0 || ( ret == -1 && errno == EINTR ) ) {
170 /* Look for and report Cygwin select() bug. */
171 if ( ret == 0 ) {
172 for ( int fd = 0; fd <= max_fd; fd++ ) {
173 if ( FD_ISSET( fd, &read_fds ) ) {
174 fprintf( stderr, "select(): nfds = 0 but read fd %d is set\n", fd );
175 }
176 }
177 }
178 /* The user should process events as usual. */
179 FD_ZERO( &read_fds );
180 ret = 0;
181 }
182
183 freeze_timestamp();
184
185 return ret;
186 }
187
188 bool read( int fd )
189#if FD_ISSET_IS_CONST
190 const
191#endif
192 {
193 assert( FD_ISSET( fd, &all_fds ) );
194 return FD_ISSET( fd, &read_fds );
195 }
196
197 /* This method consumes a signal notification. */
198 bool signal( int signum )
199 {
200 fatal_assert( signum >= 0 );
201 fatal_assert( signum <= MAX_SIGNAL_NUMBER );
202 /* XXX This requires a guard against concurrent signals. */
203 bool rv = got_signal[ signum ];
204 got_signal[ signum ] = 0;
205 return rv;
206 }
207
208 /* This method does not consume signal notifications. */
209 bool any_signal( void ) const
210 {
211 bool rv = false;
212 for (int i = 0; i < MAX_SIGNAL_NUMBER; i++) {
213 rv |= got_signal[ i ];
214 }
215 return rv;
216 }
217
218 static void set_verbose( unsigned int s_verbose ) { verbose = s_verbose; }
219
220private:
221 static const int MAX_SIGNAL_NUMBER = 64;
222 /* Number of 0-timeout selects after which we begin to think
223 * something's wrong. */
224 static const int MAX_POLLS = 10;
225
226 static void handle_signal( int signum );
227
228 int max_fd;
229
230 /* We assume writes to these ints are atomic, though we also try to mask out
231 concurrent signal handlers. */
232 int got_signal[ MAX_SIGNAL_NUMBER + 1 ];
233
234 fd_set all_fds, read_fds;
235
236 sigset_t empty_sigset;
237
238 static fd_set dummy_fd_set;
239 static sigset_t dummy_sigset;
240 int consecutive_polls;
241 static unsigned int verbose;
242};
243
244#endif
