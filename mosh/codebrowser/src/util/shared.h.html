33#ifndef SHARED_HPP
34#define SHARED_HPP
35
36#include "config.h"
37
38#ifdef HAVE_MEMORY
39#include <memory>
40#endif
41
42#ifdef HAVE_TR1_MEMORY
43#include <tr1/memory>
44#endif
45
46namespace shared {
47#ifdef HAVE_STD_SHARED_PTR
48 using std::shared_ptr;
49 using std::make_shared;
50#else
51#ifdef HAVE_STD_TR1_SHARED_PTR
52 using std::tr1::shared_ptr;
53
54 // make_shared emulation.
55 template<typename Tp>
56 inline shared_ptr<Tp>
57 make_shared()
58 {
59 return shared_ptr<Tp>(new Tp());
60 }
61 template<typename Tp, typename A1>
62 inline shared_ptr<Tp>
63 make_shared(const A1& a1)
64 {
65 return shared_ptr<Tp>(new Tp(a1));
66 }
67 template<typename Tp, typename A1, typename A2>
68 inline shared_ptr<Tp>
69 make_shared(const A1& a1, const A2& a2)
70 {
71 return shared_ptr<Tp>(new Tp(a1, a2));
72 }
73 template<typename Tp, typename A1, typename A2, typename A3>
74 inline shared_ptr<Tp>
75 make_shared(const A1& a1, const A2& a2, const A3& a3)
76 {
77 return shared_ptr<Tp>(new Tp(a1, a2, a3));
78 }
79 template<typename Tp, typename A1, typename A2, typename A3, typename A4>
80 inline shared_ptr<Tp>
81 make_shared(const A1& a1, const A2& a2, const A3& a3, const A4& a4)
82 {
83 return shared_ptr<Tp>(new Tp(a1, a2, a3, a4));
84 }
85#else
86#error Need a shared_ptr class. Try Boost::TR1.
87#endif
88#endif
89}
90#endif
