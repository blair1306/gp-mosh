59void Emulator::print( const Parser::Print *act )
60{
61 assert( act->char_present );
62
63 const wchar_t ch = act->ch;
64
65 /*
66 * Check for printing ISO 8859-1 first, it's a cheap way to detect
67 * some common narrow characters.
68 */
69 const int chwidth = ch == L'\0' ? -1 : ( Cell::isprint_iso8859_1( ch ) ? 1 : wcwidth( ch ));
70
71 Cell *this_cell = fb.get_mutable_cell();
72
73 switch ( chwidth ) {
74 case 1: /* normal character */
75 case 2: /* wide character */
76 if ( fb.ds.auto_wrap_mode && fb.ds.next_print_will_wrap ) {
77 fb.get_mutable_row( -1 )->set_wrap( true );
78 fb.ds.move_col( 0 );
79 fb.move_rows_autoscroll( 1 );
80 this_cell = NULL;
81 } else if ( fb.ds.auto_wrap_mode
82 && (chwidth == 2)
83 && (fb.ds.get_cursor_col() == fb.ds.get_width() - 1) ) {
84 /* wrap 2-cell chars if no room, even without will-wrap flag */
85 fb.reset_cell( this_cell );
86 fb.get_mutable_row( -1 )->set_wrap( false );
87 /* There doesn't seem to be a consistent way to get the
88 downstream terminal emulator to set the wrap-around
89 copy-and-paste flag on a row that ends with an empty cell
90 because a wide char was wrapped to the next line. */
91 fb.ds.move_col( 0 );
92 fb.move_rows_autoscroll( 1 );
93 this_cell = NULL;
94 }
95
96 if ( fb.ds.insert_mode ) {
97 for ( int i = 0; i < chwidth; i++ ) {
98 fb.insert_cell( fb.ds.get_cursor_row(), fb.ds.get_cursor_col() );
99 }
100 this_cell = NULL;
101 }
102
103 if (!this_cell) {
104 this_cell = fb.get_mutable_cell();
105 }
106
107 fb.reset_cell( this_cell );
108 this_cell->append( ch );
109 this_cell->set_wide( chwidth == 2 ); /* chwidth had better be 1 or 2 here */
110 fb.apply_renditions_to_cell( this_cell );
111
112 if ( chwidth == 2
113 && fb.ds.get_cursor_col() + 1 < fb.ds.get_width() ) { /* erase overlapped cell */
114 fb.reset_cell( fb.get_mutable_cell( fb.ds.get_cursor_row(), fb.ds.get_cursor_col() + 1 ) );
115 }
116
117 fb.ds.move_col( chwidth, true, true );
118
119 break;
120 case 0: /* combining character */
121 {
122 Cell *combining_cell = fb.get_combining_cell(); /* can be null if we were resized */
123 if ( combining_cell == NULL ) { /* character is now offscreen */
124 break;
125 }
126
127 if ( combining_cell->empty() ) {
128 /* cell starts with combining character */
129 /* ... but isn't necessarily the target for a new
130 base character [e.g. start of line], if the
131 combining character has been cleared with
132 a sequence like ED ("J") or EL ("K") */
133 assert( !combining_cell->get_wide() );
134 combining_cell->set_fallback( true );
135 fb.ds.move_col( 1, true, true );
136 }
137 if ( !combining_cell->full() ) {
138 combining_cell->append( ch );
139 }
140 }
141 break;
142 case -1: /* unprintable character */
143 break;
144 default:
145 assert( !"unexpected character width from wcwidth()" );
146 break;
147 }
148}

160void Emulator::Esc_dispatch( const Parser::Esc_Dispatch *act )
161{
162 /* handle 7-bit ESC-encoding of C1 control characters */
163 if ( (dispatch.get_dispatch_chars().size() == 0)
164 && (0x40 <= act->ch)
165 && (act->ch <= 0x5F) ) {
166 Parser::Esc_Dispatch act2 = *act;
167 act2.ch += 0x40;
168 dispatch.dispatch( CONTROL, &act2, &fb );
169 } else {
170 dispatch.dispatch( ESCAPE, act, &fb );
171 }
172}
