39Transition State::anywhere_rule( wchar_t ch ) const
40{
41 if ( (ch == 0x18) || (ch == 0x1A)
42 || ((0x80 <= ch) && (ch <= 0x8F))
43 || ((0x91 <= ch) && (ch <= 0x97))
44 || (ch == 0x99) || (ch == 0x9A) ) {
45 return Transition( shared::make_shared<Execute>(), &family->s_Ground );
46 } else if ( ch == 0x9C ) {
47 return Transition( &family->s_Ground );
48 } else if ( ch == 0x1B ) {
49 return Transition( &family->s_Escape );
50 } else if ( (ch == 0x98) || (ch == 0x9E) || (ch == 0x9F) ) {
51 return Transition( &family->s_SOS_PM_APC_String );
52 } else if ( ch == 0x90 ) {
53 return Transition( &family->s_DCS_Entry );
54 } else if ( ch == 0x9D ) {
55 return Transition( &family->s_OSC_String );
56 } else if ( ch == 0x9B ) {
57 return Transition( &family->s_CSI_Entry );
58 }
59
60 return Transition(( State * )NULL, ActionPointer() ); /* don't allocate an Ignore action */
61}
62
63Transition State::input( wchar_t ch ) const
64{
65 /* Check for immediate transitions. */
66 Transition anywhere = anywhere_rule( ch );
67 if ( anywhere.next_state ) {
68 anywhere.action->char_present = true;
69 anywhere.action->ch = ch;
70 return anywhere;
71 }
72 /* Normal X.364 state machine. */
73 /* Parse high Unicode codepoints like 'A'. */
74 Transition ret = this->input_state_rule( ch >= 0xA0 ? 0x41 : ch );
75 ret.action->char_present = true;
76 ret.action->ch = ch;
77 return ret;
78}

80static bool C0_prime( wchar_t ch )
81{
82 return (ch <= 0x17)
83 || (ch == 0x19)
84 || ( (0x1C <= ch) && (ch <= 0x1F) );
85}
86
87static bool GLGR ( wchar_t ch )
88{
89 return ( (0x20 <= ch) && (ch <= 0x7F) ) /* GL area */
90 || ( (0xA0 <= ch) && (ch <= 0xFF) ); /* GR area */
91}

93Transition Ground::input_state_rule( wchar_t ch ) const
94{
95 if ( C0_prime( ch ) ) {
96 return Transition( shared::make_shared< Execute >() );
97 }
98
99 if ( GLGR( ch ) ) {
100 return Transition( shared::make_shared< Print >() );
101 }
102
103 return Transition();
104}

111Transition Escape::input_state_rule( wchar_t ch ) const
112{
113 if ( C0_prime( ch ) ) {
114 return Transition( shared::make_shared< Execute >() );
115 }
116
117 if ( (0x20 <= ch) && (ch <= 0x2F) ) {
118 return Transition( shared::make_shared< Collect >(), &family->s_Escape_Intermediate );
119 }
120
121 if ( ( (0x30 <= ch) && (ch <= 0x4F) )
122 || ( (0x51 <= ch) && (ch <= 0x57) )
123 || ( ch == 0x59 )
124 || ( ch == 0x5A )
125 || ( ch == 0x5C )
126 || ( (0x60 <= ch) && (ch <= 0x7E) ) ) {
127 return Transition( shared::make_shared< Esc_Dispatch >(), &family->s_Ground );
128 }
129
130 if ( ch == 0x5B ) {
131 return Transition( &family->s_CSI_Entry );
132 }
133
134 if ( ch == 0x5D ) {
135 return Transition( &family->s_OSC_String );
136 }
137
138 if ( ch == 0x50 ) {
139 return Transition( &family->s_DCS_Entry );
140 }
141
142 if ( (ch == 0x58) || (ch == 0x5E) || (ch == 0x5F) ) {
143 return Transition( &family->s_SOS_PM_APC_String );
144 }
145
146 return Transition();
147}

171Transition CSI_Entry::input_state_rule( wchar_t ch ) const
172{
173 if ( C0_prime( ch ) ) {
174 return Transition( shared::make_shared< Execute >() );
175 }
176
177 if ( (0x40 <= ch) && (ch <= 0x7E) ) {
178 return Transition( shared::make_shared< CSI_Dispatch >(), &family->s_Ground );
179 }
180
181 if ( ( (0x30 <= ch) && (ch <= 0x39) )
182 || ( ch == 0x3B ) ) {
183 return Transition( shared::make_shared< Param >(), &family->s_CSI_Param );
184 }
185
186 if ( (0x3C <= ch) && (ch <= 0x3F) ) {
187 return Transition( shared::make_shared< Collect >(), &family->s_CSI_Param );
188 }
189
190 if ( ch == 0x3A ) {
191 return Transition( &family->s_CSI_Ignore );
192 }
193
194 if ( (0x20 <= ch) && (ch <= 0x2F) ) {
195 return Transition( shared::make_shared< Collect >(), &family->s_CSI_Intermediate );
196 }
197
198 return Transition();
199}

265Transition DCS_Entry::input_state_rule( wchar_t ch ) const
266{
267 if ( (0x20 <= ch) && (ch <= 0x2F) ) {
268 return Transition( shared::make_shared< Collect >(), &family->s_DCS_Intermediate );
269 }
270
271 if ( ch == 0x3A ) {
272 return Transition( &family->s_DCS_Ignore );
273 }
274
275 if ( ( (0x30 <= ch) && (ch <= 0x39) ) || ( ch == 0x3B ) ) {
276 return Transition( shared::make_shared< Param >(), &family->s_DCS_Param );
277 }
278
279 if ( (0x3C <= ch) && (ch <= 0x3F) ) {
280 return Transition( shared::make_shared< Collect >(), &family->s_DCS_Param );
281 }
282
283 if ( (0x40 <= ch) && (ch <= 0x7E) ) {
284 return Transition( &family->s_DCS_Passthrough );
285 }
286
287 return Transition();
288}

370Transition OSC_String::input_state_rule( wchar_t ch ) const
371{
372 if ( (0x20 <= ch) && (ch <= 0x7F) ) {
373 return Transition( shared::make_shared< OSC_Put >() );
374 }
375
376 if ( (ch == 0x9C) || (ch == 0x07) ) { /* 0x07 is xterm non-ANSI variant */
377 return Transition( &family->s_Ground );
378 }
379
380 return Transition();
381}
