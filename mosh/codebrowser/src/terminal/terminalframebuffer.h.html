54 class Renditions {
55 public:
56 typedef enum { bold, faint, italic, underlined, blink, inverse, invisible, SIZE } attribute_type;
57
58 // all together, a 32 bit word now...
59 unsigned int foreground_color : 12;
60 unsigned int background_color : 12;
61 private:
62 unsigned int attributes : 8;
63
64 public:
65 Renditions( color_type s_background );
66 void set_foreground_color( int num );
67 void set_background_color( int num );
68 void set_rendition( color_type num );
69 std::string sgr( void ) const;
70
71 bool operator==( const Renditions &x ) const
72 {
73 return ( attributes == x.attributes )
74 && ( foreground_color == x.foreground_color )
75 && ( background_color == x.background_color );
76 }
77 void set_attribute( attribute_type attr, bool val )
78 {
79 attributes = val ?
80 ( attributes | (1 << attr) ) :
81 ( attributes & ~(1 << attr) );
82 }
83 bool get_attribute( attribute_type attr ) const { return attributes & ( 1 << attr ); }
84 void clear_attributes() { attributes = 0; }
85 };

87 class Cell {
88 private:
89 typedef std::string content_type; /* can be std::string, std::vector<uint8_t>, or __gnu_cxx::__vstring */
90 content_type contents;
91 Renditions renditions;
92 unsigned int wide : 1; /* 0 = narrow, 1 = wide */

95

99 Cell( color_type background_color );
100

103 bool operator==( const Cell &x ) const
104 {
105 return ( (contents == x.contents)
106 && (fallback == x.fallback)
107 && (wide == x.wide)
108 && (renditions == x.renditions)
109 && (wrap == x.wrap) );

111
112 bool operator!=( const Cell &x ) const { return !operator==( x ); }
113
114 /* Accessors for contents field */
115 std::string debug_contents( void ) const;
116
117 bool empty( void ) const { return contents.empty(); }
118 /* 32 seems like a reasonable limit on combining characters */
119 bool full( void ) const { return contents.size() >= 32; }
120 void clear( void ) { contents.clear(); }
121
122 bool is_blank( void ) const
123 {
124 // XXX fix.
125 return ( contents.empty()
126 || contents == " "
127 || contents == "\xC2\xA0" );
128 }
129
130 bool contents_match ( const Cell &other ) const
131 {
132 return ( is_blank() && other.is_blank() )
133 || ( contents == other.contents );
134 }
135
136 bool compare( const Cell &other ) const;

138 // Is this a printing ISO 8859-1 character?
139 static bool isprint_iso8859_1( const wchar_t c )
140 {
141 return ( c <= 0xff && c >= 0xa0 ) || ( c <= 0x7e && c >= 0x20 );
142 }
143
144 static void append_to_str( std::string &dest, const wchar_t c )
145 {
146 /* ASCII? Cheat. */
147 if ( static_cast<uint32_t>(c) <= 0x7f ) {
148 dest.push_back( static_cast<char>(c) );
149 return;
150 }
151 static mbstate_t ps = mbstate_t();
152 char tmp[MB_LEN_MAX];
153 size_t ignore = wcrtomb(NULL, 0, &ps);
154 (void)ignore;
155 size_t len = wcrtomb(tmp, c, &ps);
156 dest.append( tmp, len );
157 }
158
159 void append( const wchar_t c )
160 {
161 /* ASCII? Cheat. */
162 if ( static_cast<uint32_t>(c) <= 0x7f ) {
163 contents.push_back( static_cast<char>(c) );
164 return;
165 }
166 static mbstate_t ps = mbstate_t();
167 char tmp[MB_LEN_MAX];
168 size_t ignore = wcrtomb(NULL, 0, &ps);
169 (void)ignore;
170 size_t len = wcrtomb(tmp, c, &ps);
171 contents.insert( contents.end(), tmp, tmp+len );
172 }
173
174 void print_grapheme( std::string &output ) const
175 {
176 if ( contents.empty() ) {
177 output.append( 1, ' ' );
178 return;
179 }
180 /*
181 * cells that begin with combining character get combiner
182 * attached to no-break space
183 */
184 if ( fallback ) {
185 output.append( "\xC2\xA0" );
186 }
187 output.append( contents );
188 }
189
190 /* Other accessors */
191 const Renditions& get_renditions( void ) const { return renditions; }
192 Renditions& get_renditions( void ) { return renditions; }
193 void set_renditions( const Renditions& r ) { renditions = r; }
194 bool get_wide( void ) const { return wide; }
195 void set_wide( bool w ) { wide = w; }
196 unsigned int get_width( void ) const { return wide + 1; }
197 bool get_fallback( void ) const { return fallback; }
198 void set_fallback( bool f ) { fallback = f; }
199 bool get_wrap( void ) const { return wrap; }
200 void set_wrap( bool f ) { wrap = f; }
201 };

349 class Framebuffer {
350 // To minimize copying of rows and cells, we use shared_ptr to
351 // share unchanged rows between multiple Framebuffers. If we
352 // write to a row in a Framebuffer and it is shared with other
353 // owners, we copy it first. The shared_ptr naturally manages the
354 // usage of the actual rows themselves.
355 //
356 // We gain a couple of free extras by doing this:
357 //
358 // * A quick check for equality between rows in different
359 // Framebuffers is to simply compare the pointer values. If they
360 // are equal, then the rows are obviously identical.
361 // * If no row is shared, the frame has not been modified.
362 public:
363 typedef std::vector<wchar_t> title_type;
364 typedef shared_ptr<Row> row_pointer;
365 typedef std::vector<row_pointer> rows_type; /* can be either std::vector or std::deque */
366
367 private:
368 rows_type rows;
369 title_type icon_name;
370 title_type window_title;
371 title_type clipboard;
372 unsigned int bell_count;
373 bool title_initialized; /* true if the window title has been set via an OSC */
374
375 row_pointer newrow( void )
376 {
377 const size_t w = ds.get_width();
378 const color_type c = ds.get_background_rendition();
379 return make_shared<Row>( w, c );
380 }
381
382 public:
383 Framebuffer( int s_width, int s_height );
384 Framebuffer( const Framebuffer &other );
385 Framebuffer &operator=( const Framebuffer &other );
386 DrawState ds;
387
388 const rows_type &get_rows() const { return rows; }
389
390 void scroll( int N );
391 void move_rows_autoscroll( int rows );
392
393 inline const Row *get_row( int row ) const
394 {
395 if ( row == -1 ) row = ds.get_cursor_row();
396
397 return rows.at( row ).get();
398 }
399
400 inline const Cell *get_cell( int row = -1, int col = -1 ) const
401 {
402 if ( row == -1 ) row = ds.get_cursor_row();
403 if ( col == -1 ) col = ds.get_cursor_col();
404
405 return &rows.at( row )->cells.at( col );
406 }
407
408 Row *get_mutable_row( int row )
409 {
410 if ( row == -1 ) row = ds.get_cursor_row();
411 row_pointer &mutable_row = rows.at( row );
412 // If the row is shared, copy it.
413 if (!mutable_row.unique()) {
414 mutable_row = make_shared<Row>( *mutable_row );
415 }
416 return mutable_row.get();
417 }
418
419 Cell *get_mutable_cell( int row = -1, int col = -1 )
420 {
421 if ( row == -1 ) row = ds.get_cursor_row();
422 if ( col == -1 ) col = ds.get_cursor_col();
423
424 return &get_mutable_row( row )->cells.at( col );
425 }

442 void set_icon_name( const title_type &s ) { icon_name = s; }
443 void set_window_title( const title_type &s ) { window_title = s; }
444 void set_clipboard( const title_type &s ) { clipboard = s; }

459 bool operator==( const Framebuffer &x ) const
460 {
461 return ( rows == x.rows ) && ( window_title == x.window_title ) && ( clipboard == x.clipboard ) && ( bell_count == x.bell_count ) && ( ds == x.ds );
462 }
