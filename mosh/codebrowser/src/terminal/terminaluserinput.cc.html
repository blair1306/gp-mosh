33#include <assert.h>
34#include "terminaluserinput.h"
35
36using namespace Terminal;
37using namespace std;
38
39string UserInput::input( const Parser::UserByte *act,
40 bool application_mode_cursor_keys )
41{
42 /* The user will always be in application mode. If stm is not in
43 application mode, convert user's cursor control function to an
44 ANSI cursor control sequence */
45
46 /* We need to look ahead one byte in the SS3 state to see if
47 the next byte will be A, B, C, or D (cursor control keys). */
48
49 /* This doesn't handle the 8-bit SS3 C1 control, which would be
50 two octets in UTF-8. Fortunately nobody seems to send this. */
51
52 switch ( state ) {
53 case Ground:
54 if ( act->c == 0x1b ) { /* ESC */
55 state = ESC;
56 }
57 return string( &act->c, 1 );
58
59 case ESC:
60 if ( act->c == 'O' ) { /* ESC O = 7-bit SS3 */
61 state = SS3;
62 return string();
63 }
64 state = Ground;
65 return string( &act->c, 1 );
66
67 case SS3:
68 state = Ground;
69 if ( (!application_mode_cursor_keys)
70 && (act->c >= 'A')
71 && (act->c <= 'D') ) {
72 char translated_cursor[ 2 ] = { '[', act->c };
73 return string( translated_cursor, 2 );
74 } else {
75 char original_cursor[ 2 ] = { 'O', act->c };
76 return string( original_cursor, 2 );
77 }
78
79 default:
80 assert( !"unexpected state" );
81 state = Ground;
82 return string();
83 }
84}
