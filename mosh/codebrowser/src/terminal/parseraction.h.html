41namespace Terminal {
42 class Emulator;
43}
44
45namespace Parser {
46 class Action
47 {
48 public:
49 wchar_t ch;
50 bool char_present;
51
52 virtual std::string name( void ) = 0;
53
54 virtual void act_on_terminal( Terminal::Emulator * ) const {};
55
56 virtual bool ignore() const { return false; }
57
58 Action() : ch( -1 ), char_present( false ) {};
59 virtual ~Action() {};
60 };
61
62 typedef shared::shared_ptr<Action> ActionPointer;
63 typedef std::vector<ActionPointer> Actions;

130 class UserByte : public Action {
131 /* user keystroke -- not part of the host-source state machine*/
132 public:
133 char c; /* The user-source byte. We don't try to interpret the charset */
134
135 std::string name( void ) { return std::string( "UserByte" ); }
136 void act_on_terminal( Terminal::Emulator *emu ) const;
137
138 UserByte( int s_c ) : c( s_c ) {}
139
140 bool operator==( const UserByte &other ) const
141 {
142 return c == other.c;
143 }
144 };
145
146 class Resize : public Action {
147 /* resize event -- not part of the host-source state machine*/
148 public:
149 size_t width, height;
150
151 std::string name( void ) { return std::string( "Resize" ); }
152 void act_on_terminal( Terminal::Emulator *emu ) const;
153
154 Resize( size_t s_width, size_t s_height )
155 : width( s_width ),
156 height( s_height )
157 {}
158
159 bool operator==( const Resize &other ) const
160 {
161 return ( width == other.width ) && ( height == other.height );
162 }
163 };
