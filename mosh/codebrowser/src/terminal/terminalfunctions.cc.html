47static void clearline( Framebuffer *fb, int row, int start, int end )
48{
49 for ( int col = start; col <= end; col++ ) {
50 fb->reset_cell( fb->get_mutable_cell( row, col ) );
51 }
52}

54/* erase in line */
55static void CSI_EL( Framebuffer *fb, Dispatcher *dispatch )
56{
57 switch ( dispatch->getparam( 0, 0 ) ) {
58 case 0: /* default: active position to end of line, inclusive */
59 clearline( fb, -1, fb->ds.get_cursor_col(), fb->ds.get_width() - 1 );
60 break;
61 case 1: /* start of screen to active position, inclusive */
62 clearline( fb, -1, 0, fb->ds.get_cursor_col() );
63 break;
64 case 2: /* all of line */
65 fb->reset_row( fb->get_mutable_row( -1 ) );
66 break;
67 default:
68 break;
69 }
70}

74/* erase in display */
75static void CSI_ED( Framebuffer *fb, Dispatcher *dispatch ) {
76 switch ( dispatch->getparam( 0, 0 ) ) {
77 case 0: /* active position to end of screen, inclusive */
78 clearline( fb, -1, fb->ds.get_cursor_col(), fb->ds.get_width() - 1 );
79 for ( int y = fb->ds.get_cursor_row() + 1; y < fb->ds.get_height(); y++ ) {
80 fb->reset_row( fb->get_mutable_row( y ) );
81 }
82 break;
83 case 1: /* start of screen to active position, inclusive */
84 for ( int y = 0; y < fb->ds.get_cursor_row(); y++ ) {
85 fb->reset_row( fb->get_mutable_row( y ) );
86 }
87 clearline( fb, -1, 0, fb->ds.get_cursor_col() );
88 break;
89 case 2: /* entire screen */
90 for ( int y = 0; y < fb->ds.get_height(); y++ ) {
91 fb->reset_row( fb->get_mutable_row( y ) );
92 }
93 break;
94 default:
95 break;
96 }
97}

101/* cursor movement -- relative and absolute */
102static void CSI_cursormove( Framebuffer *fb, Dispatcher *dispatch )
103{
104 int num = dispatch->getparam( 0, 1 );
105
106 switch ( dispatch->get_dispatch_chars()[ 0 ] ) {
107 case 'A':
108 fb->ds.move_row( -num, true );
109 break;
110 case 'B':
111 fb->ds.move_row( num, true );
112 break;
113 case 'C':
114 fb->ds.move_col( num, true );
115 break;
116 case 'D':
117 fb->ds.move_col( -num, true );
118 break;
119 case 'H':
120 case 'f':
121 fb->ds.move_row( dispatch->getparam( 0, 1 ) - 1 );
122 fb->ds.move_col( dispatch->getparam( 1, 1 ) - 1 );
123 break;
124 default:
125 break;
126 }
127}

210/* horizontal tab */
211static void HT_n( Framebuffer *fb, size_t count )
212{
213 int col = fb->ds.get_next_tab( count );
214 if ( col == -1 ) { /* no tabs, go to end of line */
215 col = fb->ds.get_width() - 1;
216 }
217
218 /* A horizontal tab is the only operation that preserves but
219 does not set the wrap state. It also starts a new grapheme. */
220
221 bool wrap_state_save = fb->ds.next_print_will_wrap;
222 fb->ds.move_col( col, false );
223 fb->ds.next_print_will_wrap = wrap_state_save;
224}
225
226static void Ctrl_HT( Framebuffer *fb, Dispatcher *dispatch __attribute((unused)) )
227{
228 HT_n( fb, 1 );
229}

255/* tabulation clear */
256static void CSI_TBC( Framebuffer *fb, Dispatcher *dispatch )
257{
258 int param = dispatch->getparam( 0, 0 );
259 switch ( param ) {
260 case 0: /* clear this tab stop */
261 fb->ds.clear_tab( fb->ds.get_cursor_col() );
262 break;
263 case 3: /* clear all tab stops */
264 fb->ds.clear_default_tabs();
265 for ( int x = 0; x < fb->ds.get_width(); x++ ) {
266 fb->ds.clear_tab( x );
267 }
268 break;
269 default:
270 break;
271 }
272}

277static bool *get_DEC_mode( int param, Framebuffer *fb ) {
278 switch ( param ) {
279 case 1: /* cursor key mode */
280 return &(fb->ds.application_mode_cursor_keys);
281 case 3: /* 80/132. Ignore but clear screen. */
282 /* clear screen */
283 fb->ds.move_row( 0 );
284 fb->ds.move_col( 0 );
285 for ( int y = 0; y < fb->ds.get_height(); y++ ) {
286 fb->reset_row( fb->get_mutable_row( y ) );
287 }
288 return NULL;
289 case 5: /* reverse video */
290 return &(fb->ds.reverse_video);
291 case 6: /* origin */
292 fb->ds.move_row( 0 );
293 fb->ds.move_col( 0 );
294 return &(fb->ds.origin_mode);
295 case 7: /* auto wrap */
296 return &(fb->ds.auto_wrap_mode);
297 case 25:
298 return &(fb->ds.cursor_visible);
299 case 1004: /* xterm mouse focus event */
300 return &(fb->ds.mouse_focus_event);
301 case 1007: /* xterm mouse alternate scroll */
302 return &(fb->ds.mouse_alternate_scroll);
303 case 2004: /* bracketed paste */
304 return &(fb->ds.bracketed_paste);
305 default:
306 break;
307 }
308 return NULL;
309}

317/* set private mode */
318static void CSI_DECSM( Framebuffer *fb, Dispatcher *dispatch )
319{
320 for ( int i = 0; i < dispatch->param_count(); i++ ) {
321 int param = dispatch->getparam( i, 0 );
322 if (param == 9 || (param >= 1000 && param <= 1003)) {
323 fb->ds.mouse_reporting_mode = (Terminal::DrawState::MouseReportingMode) param;
324 } else if (param == 1005 || param == 1006 || param == 1015) {
325 fb->ds.mouse_encoding_mode = (Terminal::DrawState::MouseEncodingMode) param;
326 } else {
327 set_if_available( get_DEC_mode( param, fb ), true );
328 }
329 }
330}
331
332/* clear private mode */
333static void CSI_DECRM( Framebuffer *fb, Dispatcher *dispatch )
334{
335 for ( int i = 0; i < dispatch->param_count(); i++ ) {
336 int param = dispatch->getparam( i, 0 );
337 if (param == 9 || (param >= 1000 && param <= 1003)) {
338 fb->ds.mouse_reporting_mode = Terminal::DrawState::MOUSE_REPORTING_NONE;
339 } else if (param == 1005 || param == 1006 || param == 1015) {
340 fb->ds.mouse_encoding_mode = Terminal::DrawState::MOUSE_ENCODING_DEFAULT;
341 } else {
342 set_if_available( get_DEC_mode( param, fb ), false );
343 }
344 }
345}

383/* set top and bottom margins */
384static void CSI_DECSTBM( Framebuffer *fb, Dispatcher *dispatch )
385{
386 int top = dispatch->getparam( 0, 1 );
387 int bottom = dispatch->getparam( 1, fb->ds.get_height() );
388
389 if ( (bottom <= top)
390 || (top > fb->ds.get_height())
391 || (top == 0 && bottom == 1) ) {
392 return; /* invalid, xterm ignores */
393 }
394
395 fb->ds.set_scrolling_region( top - 1, bottom - 1 );
396 fb->ds.move_row( 0 );
397 fb->ds.move_col( 0 );
398}

409/* select graphics rendition -- e.g., bold, blinking, etc. */
410static void CSI_SGR( Framebuffer *fb, Dispatcher *dispatch )
411{
412 for ( int i = 0; i < dispatch->param_count(); i++ ) {
413 int rendition = dispatch->getparam( i, 0 );
414 /* We need to special-case the handling of [34]8 ; 5 ; Ps,
415 because Ps of 0 in that case does not mean reset to default, even
416 though it means that otherwise (as usually renditions are applied
417 in order). */
418 if ((rendition == 38 || rendition == 48) &&
419 (dispatch->param_count() - i >= 3) &&
420 (dispatch->getparam( i+1, -1 ) == 5)) {
421 (rendition == 38) ?
422 fb->ds.set_foreground_color( dispatch->getparam( i+2, 0 ) ) :
423 fb->ds.set_background_color( dispatch->getparam( i+2, 0 ) );
424 i += 2;
425 continue;
426 }
427 fb->ds.add_rendition( rendition );
428 }
429}

448static void CSI_DSR( Framebuffer *fb, Dispatcher *dispatch )

571/* xterm uses an Operating System Command to set the window title */
572void Dispatcher::OSC_dispatch( const Parser::OSC_End *act __attribute((unused)), Framebuffer *fb )
573{
574 /* handle osc copy clipboard sequence 52;c; */
575 if ( OSC_string.size() >= 5 && OSC_string[ 0 ] == L'5' &&
576 OSC_string[ 1 ] == L'2' && OSC_string[ 2 ] == L';' &&
577 OSC_string[ 3 ] == L'c' && OSC_string[ 4 ] == L';') {
578 Terminal::Framebuffer::title_type clipboard(
579 OSC_string.begin() + 5, OSC_string.end() );
580 fb->set_clipboard( clipboard );
581 /* handle osc terminal title sequence */
582 } else if ( OSC_string.size() >= 1 ) {
583 long cmd_num = -1;
584 int offset = 0;
585 if ( OSC_string[ 0 ] == L';' ) {
586 /* OSC of the form "\033];<title>\007" */
587 cmd_num = 0; /* treat it as as a zero */
588 offset = 1;
589 } else if ( (OSC_string.size() >= 2) && (OSC_string[ 1 ] == L';') ) {
590 /* OSC of the form "\033]X;<title>\007" where X can be:
591 * 0: set icon name and window title
592 * 1: set icon name
593 * 2: set window title */
594 cmd_num = OSC_string[ 0 ] - L'0';
595 offset = 2;
596 }
597 bool set_icon = cmd_num == 0 || cmd_num == 1;
598 bool set_title = cmd_num == 0 || cmd_num == 2;
599 if ( set_icon || set_title ) {
600 fb->set_title_initialized();
601 int title_length = min(OSC_string.size(), (size_t)256);
602 Terminal::Framebuffer::title_type newtitle( OSC_string.begin() + offset, OSC_string.begin() + title_length );
603 if ( set_icon ) { fb->set_icon_name( newtitle ); }
604 if ( set_title ) { fb->set_window_title( newtitle ); }
605 }
606 }
607}
