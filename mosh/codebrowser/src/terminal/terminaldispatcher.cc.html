45static const size_t MAXIMUM_CLIPBOARD_SIZE = 16*1024;
46
47Dispatcher::Dispatcher()
48 : params(), parsed_params(), parsed( false ), dispatch_chars(),
49 OSC_string(), terminal_to_host()
50{}

52void Dispatcher::newparamchar( const Parser::Param *act )
53{
54 assert( act->char_present );
55 assert( (act->ch == ';') || ( (act->ch >= '0') && (act->ch <= '9') ) );
56 if ( params.length() < 100 ) {
57 /* enough for 16 five-char params plus 15 semicolons */
58 params.push_back( act->ch );
59 }
60 parsed = false;
61}
62
63void Dispatcher::collect( const Parser::Collect *act )
64{
65 assert( act->char_present );
66 if ( ( dispatch_chars.length() < 8 ) /* never should need more than 2 */
67 && ( act->ch <= 255 ) ) { /* ignore non-8-bit */
68 dispatch_chars.push_back( act->ch );
69 }
70}
71
72void Dispatcher::clear( const Parser::Clear *act __attribute((unused)) )
73{
74 params.clear();
75 dispatch_chars.clear();
76 parsed = false;
77}

79void Dispatcher::parse_params( void )
80{
81 if ( parsed ) {
82 return;
83 }
84
85 parsed_params.clear();
86 const char *str = params.c_str();
87 const char *segment_begin = str;
88
89 while ( 1 ) {
90 const char *segment_end = strchr( segment_begin, ';' );
91 if ( segment_end == NULL ) {
92 break;
93 }
94
95 errno = 0;
96 char *endptr;
97 long val = strtol( segment_begin, &endptr, 10 );
98 if ( endptr == segment_begin ) {
99 val = -1;
100 }
101
102 if ( val > PARAM_MAX || errno == ERANGE ) {
103 val = -1;
104 errno = 0;
105 }
106
107 if ( errno == 0 || segment_begin == endptr ) {
108 parsed_params.push_back( val );
109 }
110
111 segment_begin = segment_end + 1;
112 }
113
114 /* get last param */
115 errno = 0;
116 char *endptr;
117 long val = strtol( segment_begin, &endptr, 10 );
118 if ( endptr == segment_begin ) {
119 val = -1;
120 }
121
122 if ( val > PARAM_MAX || errno == ERANGE ) {
123 val = -1;
124 errno = 0;
125 }
126
127 if ( errno == 0 || segment_begin == endptr ) {
128 parsed_params.push_back( val );
129 }
130
131 parsed = true;
132}

134int Dispatcher::getparam( size_t N, int defaultval )
135{
136 int ret = defaultval;
137 if ( !parsed ) {
138 parse_params();
139 }
140
141 if ( parsed_params.size() > N ) {
142 ret = parsed_params[ N ];
143 }
144
145 if ( ret < 1 ) ret = defaultval;
146
147 return ret;
148}
149
150int Dispatcher::param_count( void )
151{
152 if ( !parsed ) {
153 parse_params();
154 }
155
156 return parsed_params.size();
157}

159std::string Dispatcher::str( void )
160{
161 char assum[ 64 ];
162 snprintf( assum, 64, "[dispatch=\"%s\" params=\"%s\"]",
163 dispatch_chars.c_str(), params.c_str() );
164 return std::string( assum );
165}

168DispatchRegistry & Terminal::get_global_dispatch_registry( void )
169{
170 static DispatchRegistry global_dispatch_registry;
171 return global_dispatch_registry;
172}

174static void register_function( Function_Type type,
175 const std::string & dispatch_chars,
176 Function f )
177{
178 switch ( type ) {
179 case ESCAPE:
180 get_global_dispatch_registry().escape.insert( dispatch_map_t::value_type( dispatch_chars, f ) );
181 break;
182 case CSI:
183 get_global_dispatch_registry().CSI.insert( dispatch_map_t::value_type( dispatch_chars, f ) );
184 break;
185 case CONTROL:
186 get_global_dispatch_registry().control.insert( dispatch_map_t::value_type( dispatch_chars, f ) );
187 break;
188 }
189}

191Function::Function( Function_Type type, const std::string & dispatch_chars,
192 void (*s_function)( Framebuffer *, Dispatcher * ),
193 bool s_clears_wrap_state )
194 : function( s_function ), clears_wrap_state( s_clears_wrap_state )
195{
196 register_function( type, dispatch_chars, *this );
197}

199void Dispatcher::dispatch( Function_Type type, const Parser::Action *act, Framebuffer *fb )
200{
201 /* add final char to dispatch key */
202 if ( (type == ESCAPE) || (type == CSI) ) {
203 assert( act->char_present );
204 Parser::Collect act2;
205 act2.char_present = true;
206 act2.ch = act->ch;
207 collect( &act2 );
208 }
209
210 dispatch_map_t *map = NULL;
211 switch ( type ) {
212 case ESCAPE: map = &get_global_dispatch_registry().escape; break;
213 case CSI: map = &get_global_dispatch_registry().CSI; break;
214 case CONTROL: map = &get_global_dispatch_registry().control; break;
215 }
216
217 std::string key = dispatch_chars;
218 if ( type == CONTROL ) {
219 assert( act->ch <= 255 );
220 char ctrlstr[ 2 ] = { (char)act->ch, 0 };
221 key = std::string( ctrlstr, 1 );
222 }
223
224 dispatch_map_t::const_iterator i = map->find( key );
225 if ( i == map->end() ) {
226 /* unknown function */
227 fb->ds.next_print_will_wrap = false;
228 return;
229 }
230 if ( i->second.clears_wrap_state ) {
231 fb->ds.next_print_will_wrap = false;
232 }
233 i->second.function( fb, this );
234}

236void Dispatcher::OSC_put( const Parser::OSC_Put *act )
237{
238 assert( act->char_present );
239 if ( OSC_string.size() < MAXIMUM_CLIPBOARD_SIZE) {
240 OSC_string.push_back( act->ch );
241 }
242}
243
244void Dispatcher::OSC_start( const Parser::OSC_Start *act __attribute((unused)) )
245{
246 OSC_string.clear();
247}

249bool Dispatcher::operator==( const Dispatcher &x ) const
250{
251 return ( params == x.params )
252 && ( parsed_params == x.parsed_params )
253 && ( parsed == x.parsed )
254 && ( dispatch_chars == x.dispatch_chars )
255 && ( OSC_string == x.OSC_string )
256 && ( terminal_to_host == x.terminal_to_host );
257}
