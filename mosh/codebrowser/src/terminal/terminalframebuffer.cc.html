111void Framebuffer::scroll( int N )
112{
113 if ( N >= 0 ) {
114 delete_line( ds.get_scrolling_region_top_row(), N );
115 } else {
116 insert_line( ds.get_scrolling_region_top_row(), -N );
117 }

126void DrawState::snap_cursor_to_border( void )
127{
128 if ( cursor_row < limit_top() ) cursor_row = limit_top();
129 if ( cursor_row > limit_bottom() ) cursor_row = limit_bottom();
130 if ( cursor_col < 0 ) cursor_col = 0;
131 if ( cursor_col >= width ) cursor_col = width - 1;
132}

134void DrawState::move_row( int N, bool relative )
135{
136 if ( relative ) {
137 cursor_row += N;
138 } else {
139 cursor_row = N + limit_top();
140 }
141
142 snap_cursor_to_border();
143 new_grapheme();
144 next_print_will_wrap = false;
145}
146
147void DrawState::move_col( int N, bool relative, bool implicit )
148{
149 if ( implicit ) {
150 new_grapheme();
151 }
152
153 if ( relative ) {
154 cursor_col += N;
155 } else {
156 cursor_col = N;
157 }
158
159 if ( implicit ) {
160 next_print_will_wrap = (cursor_col >= width);
161 }
162
163 snap_cursor_to_border();
164 if ( !implicit ) {
165 new_grapheme();
166 next_print_will_wrap = false;
167 }
168}

170void Framebuffer::move_rows_autoscroll( int rows )
171{
172 /* don't scroll if outside the scrolling region */
173 if ( (ds.get_cursor_row() < ds.get_scrolling_region_top_row())
174 || (ds.get_cursor_row() > ds.get_scrolling_region_bottom_row()) ) {
175 ds.move_row( rows, true );
176 return;
177 }
178
179 if ( ds.get_cursor_row() + rows > ds.get_scrolling_region_bottom_row() ) {
180 int N = ds.get_cursor_row() + rows - ds.get_scrolling_region_bottom_row();
181 scroll( N );
182 ds.move_row( -N, true );
183 } else if ( ds.get_cursor_row() + rows < ds.get_scrolling_region_top_row() ) {
184 int N = ds.get_cursor_row() + rows - ds.get_scrolling_region_top_row();
185 scroll( N );
186 ds.move_row( -N, true );
187 }
188
189 ds.move_row( rows, true );
190}

192Cell *Framebuffer::get_combining_cell( void )
193{
194 if ( (ds.get_combining_char_col() < 0)
195 || (ds.get_combining_char_row() < 0)
196 || (ds.get_combining_char_col() >= ds.get_width())
197 || (ds.get_combining_char_row() >= ds.get_height()) ) {
198 return NULL;
199 } /* can happen if a resize came in between */
200
201 return get_mutable_cell( ds.get_combining_char_row(), ds.get_combining_char_col() );
202}

214int DrawState::get_next_tab( int count ) const
215{
216 if ( count >= 0 ) {
217 for ( int i = cursor_col + 1; i < width; i++ ) {
218 if ( tabs[ i ] && --count == 0 ) {
219 return i;
220 }
221 }
222 return -1;
223 }
224 for ( int i = cursor_col - 1; i > 0; i-- ) {
225 if ( tabs[ i ] && ++count == 0 ) {
226 return i;
227 }
228 }
229 return 0;
230}

232void DrawState::set_scrolling_region( int top, int bottom )
233{
234 if ( height < 1 ) {
235 return;
236 }
237
238 scrolling_region_top_row = top;
239 scrolling_region_bottom_row = bottom;
240
241 if ( scrolling_region_top_row < 0 ) scrolling_region_top_row = 0;
242 if ( scrolling_region_bottom_row >= height ) scrolling_region_bottom_row = height - 1;
243
244 if ( scrolling_region_bottom_row < scrolling_region_top_row )
245 scrolling_region_bottom_row = scrolling_region_top_row;
246 /* real rule requires TWO-line scrolling region */
247
248 if ( origin_mode ) {
249 snap_cursor_to_border();
250 new_grapheme();
251 }
252}
253
254int DrawState::limit_top( void ) const
255{
256 return origin_mode ? scrolling_region_top_row : 0;
257}
258
259int DrawState::limit_bottom( void ) const
260{
261 return origin_mode ? scrolling_region_bottom_row : height - 1;
262}

300void Framebuffer::insert_line( int before_row, int count )
301{
302 if ( (before_row < ds.get_scrolling_region_top_row())
303 || (before_row > ds.get_scrolling_region_bottom_row() + 1) ) {
304 return;
305 }
306
307 int scroll = ds.get_scrolling_region_bottom_row() + 1 - before_row;
308 if ( count < scroll ) {
309 scroll = count;
310 }
311
312 if ( scroll == 0 ) {
313 return;
314 }
315
316 // delete old rows
317 rows_type::iterator start = rows.begin() + ds.get_scrolling_region_bottom_row() + 1 - scroll;
318 rows.erase( start, start + scroll );
319 // insert new rows
320 start = rows.begin() + before_row;
321 rows.insert( start, scroll, newrow());
322}
323
324void Framebuffer::delete_line( int row, int count )
325{
326 if ( (row < ds.get_scrolling_region_top_row())
327 || (row > ds.get_scrolling_region_bottom_row()) ) {
328 return;
329 }
330
331 int scroll = ds.get_scrolling_region_bottom_row() + 1 - row;
332 if ( count < scroll ) {
333 scroll = count;
334 }
335
336 if ( scroll == 0 ) {
337 return;
338 }
339
340 // delete old rows
341 rows_type::iterator start = rows.begin() + row;
342 rows.erase( start, start + scroll );
343 // insert a block of dummy rows
344 start = rows.begin() + ds.get_scrolling_region_bottom_row() + 1 - scroll;
345 rows.insert( start, scroll, newrow());
346}

401void Framebuffer::resize( int s_width, int s_height )
402{
403 assert( s_width > 0 );
404 assert( s_height > 0 );
405
406 int oldheight = ds.get_height();
407 int oldwidth = ds.get_width();
408 ds.resize( s_width, s_height );
409
410 row_pointer blankrow( newrow());
411 if ( oldheight != s_height ) {
412 rows.resize( s_height, blankrow );
413 }
414 if (oldwidth == s_width) {
415 return;
416 }
417 for ( rows_type::iterator i = rows.begin();
418 i != rows.end() && *i != blankrow;
419 i++ ) {
420 *i = make_shared<Row>( **i );
421 (*i)->set_wrap( false );
422 (*i)->cells.resize( s_width, Cell( ds.get_background_rendition() ) );
423 }
424}

426void DrawState::resize( int s_width, int s_height )
427{
428 if ( (width != s_width)
429 || (height != s_height) ) {
430 /* reset entire scrolling region on any resize */
431 /* xterm and rxvt-unicode do this. gnome-terminal only
432 resets scrolling region if it has to become smaller in resize */
433 scrolling_region_top_row = 0;
434 scrolling_region_bottom_row = s_height - 1;
435 }
436
437 tabs.resize( s_width );
438 if ( default_tabs ) {
439 reinitialize_tabs( width );
440 }
441
442 width = s_width;
443 height = s_height;
444
445 snap_cursor_to_border();
446
447 /* saved cursor will be snapped to border on restore */
448
449 /* invalidate combining char cell if necessary */
450 if ( (combining_char_col >= width)
451 || (combining_char_row >= height) ) {
452 combining_char_col = combining_char_row = -1;
453 }
454}

461/* This routine cannot be used to set a color beyond the 16-color set. */
462void Renditions::set_rendition( color_type num )
463{
464 if ( num == 0 ) {
465 clear_attributes();
466 foreground_color = background_color = 0;
467 return;
468 }
469
470 if ( num == 39 ) {
471 foreground_color = 0;
472 return;
473 } else if ( num == 49 ) {
474 background_color = 0;
475 return;
476 }
477
478 if ( (30 <= num) && (num <= 37) ) { /* foreground color in 8-color set */
479 foreground_color = num;
480 return;
481 } else if ( (40 <= num) && (num <= 47) ) { /* background color in 8-color set */
482 background_color = num;
483 return;
484 } else if ( (90 <= num) && (num <= 97) ) { /* foreground color in 16-color set */
485 foreground_color = num - 90 + 38;
486 return;
487 } else if ( (100 <= num) && (num <= 107) ) { /* background color in 16-color set */
488 background_color = num - 100 + 48;
489 return;
490 }
491
492 bool value = num < 9;
493 switch ( num ) {
494 case 1: case 22: set_attribute(bold, value); break;
495 case 3: case 23: set_attribute(italic, value); break;
496 case 4: case 24: set_attribute(underlined, value); break;
497 case 5: case 25: set_attribute(blink, value); break;
498 case 7: case 27: set_attribute(inverse, value); break;
499 case 8: case 28: set_attribute(invisible, value); break;
500 default: break; /* ignore unknown rendition */
501 }
502}

518std::string Renditions::sgr( void ) const
519{
520 std::string ret;
521
522 ret.append( "\033[0" );
523 if ( get_attribute( bold ) ) ret.append( ";1" );
524 if ( get_attribute( italic ) ) ret.append( ";3" );
525 if ( get_attribute( underlined ) ) ret.append( ";4" );
526 if ( get_attribute( blink ) ) ret.append( ";5" );
527 if ( get_attribute( inverse ) ) ret.append( ";7" );
528 if ( get_attribute( invisible ) ) ret.append( ";8" );
529
530 if ( foreground_color
531 && (foreground_color <= 37) ) {
532 /* ANSI foreground color */
533 char col[ 8 ];
534 snprintf( col, 8, ";%d", foreground_color );
535 ret.append( col );
536 }
537
538 if ( background_color
539 && (background_color <= 47) ) {
540 char col[ 8 ];
541 snprintf( col, 8, ";%d", background_color );
542 ret.append( col );
543 }
544
545 ret.append( "m" );
546
547 if ( foreground_color > 37 ) { /* use 256-color set */
548 char col[ 64 ];
549 snprintf( col, 64, "\033[38;5;%dm", foreground_color - 30 );
550 ret.append( col );
551 }
552
553 if ( background_color > 47 ) { /* use 256-color set */
554 char col[ 64 ];
555 snprintf( col, 64, "\033[48;5;%dm", background_color - 40 );
556 ret.append( col );
557 }
558
559 return ret;
560}

581std::string Cell::debug_contents( void ) const
582{
583 if ( contents.empty() ) {
584 return "'_' ()";
585 }
586 std::string chars( 1, '\'' );
587 print_grapheme( chars );
588 chars.append( "' [" );
589 const char *lazycomma = "";
590 char buf[64];
591 for ( content_type::const_iterator i = contents.begin();
592 i < contents.end();
593 i++ ) {
594
595 snprintf( buf, sizeof buf, "%s0x%02x", lazycomma, static_cast<uint8_t>(*i) );
596 chars.append( buf );
597 lazycomma = ", ";
598 }
599 chars.append( "]" );
600 return chars;
601}

603bool Cell::compare( const Cell &other ) const
604{
605 bool ret = false;
606
607 std::string grapheme, other_grapheme;
608
609 print_grapheme( grapheme );
610 other.print_grapheme( other_grapheme );
611
612 if ( grapheme != other_grapheme ) {
613 ret = true;
614 fprintf( stderr, "Graphemes: '%s' vs. '%s'\n",
615 grapheme.c_str(), other_grapheme.c_str() );
616 }
617
618 if ( !contents_match( other ) ) {
619 // ret = true;
620 fprintf( stderr, "Contents: %s (%ld) vs. %s (%ld)\n",
621 debug_contents().c_str(),
622 static_cast<long int>( contents.size() ),
623 other.debug_contents().c_str(),
624 static_cast<long int>( other.contents.size() ) );
625 }
626
627 if ( fallback != other.fallback ) {
628 // ret = true;
629 fprintf( stderr, "fallback: %d vs. %d\n",
630 fallback, other.fallback );
631 }
632
633 if ( wide != other.wide ) {
634 ret = true;
635 fprintf( stderr, "width: %d vs. %d\n",
636 wide, other.wide );
637 }
638
639 if ( !(renditions == other.renditions) ) {
640 ret = true;
641 fprintf( stderr, "renditions differ\n" );
642 }
643
644 if ( wrap != other.wrap ) {
645 ret = true;
646 fprintf( stderr, "wrap: %d vs. %d\n",
647 wrap, other.wrap );
648 }
649
650 return ret;
651}
