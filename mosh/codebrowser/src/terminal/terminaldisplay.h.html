33#ifndef TERMINALDISPLAY_HPP
34#define TERMINALDISPLAY_HPP
35
36#include "terminalframebuffer.h"
37
38namespace Terminal {
39 /* variables used within a new_frame */
40 class FrameState {
41 public:
42 std::string str;
43
44 int cursor_x, cursor_y;
45 Renditions current_rendition;
46 bool cursor_visible;
47
48 const Framebuffer &last_frame;
49
50 FrameState( const Framebuffer &s_last );
51
52 void append( char c ) { str.append( 1, c ); }
53 void append( size_t s, char c ) { str.append( s, c ); }
54 void append( wchar_t wc ) { Cell::append_to_str( str, wc ); }
55 void append( const char * s ) { str.append( s ); }
56 void append_string( const std::string &append ) { str.append(append); }
57
58 void append_cell(const Cell & cell) { cell.print_grapheme( str ); }
59 void append_silent_move( int y, int x );
60 void append_move( int y, int x );
61 void update_rendition( const Renditions &r, bool force = false );
62 };
63
64 class Display {
65 private:
66 bool has_ech; /* erase character is part of vt200 but not supported by tmux
67 (or by "screen" terminfo entry, which is what tmux advertises) */
68
69 bool has_bce; /* erases result in cell filled with background color */
70
71 bool has_title; /* supports window title and icon name */
72
73 const char *smcup, *rmcup; /* enter and exit alternate screen mode */
74
75 bool put_row( bool initialized, FrameState &frame, const Framebuffer &f, int frame_y, const Row &old_row, bool wrap ) const;
76
77 public:
78 std::string open() const;
79 std::string close() const;
80
81 std::string new_frame( bool initialized, const Framebuffer &last, const Framebuffer &f ) const;
82
83 Display( bool use_environment );
84 };
85}
