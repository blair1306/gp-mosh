40/* Print a new "frame" to the terminal, using ANSI/ECMA-48 escape codes. */

42static const Renditions & initial_rendition( void )
43{
44 const static Renditions blank = Renditions( 0 );
45 return blank;
46}

61std::string Display::new_frame( bool initialized, const Framebuffer &last, const Framebuffer &f ) const
62{
63 FrameState frame( last );
64
65 char tmp[ 64 ];
66
67 /* has bell been rung? */
68 if ( f.get_bell_count() != frame.last_frame.get_bell_count() ) {
69 frame.append( '\007' );
70 }
71 typedef Terminal::Framebuffer::title_type title_type;
72
73 /* has icon name or window title changed? */
74 if ( has_title && f.is_title_initialized() &&
75 ( (!initialized)
76 || (f.get_icon_name() != frame.last_frame.get_icon_name())
77 || (f.get_window_title() != frame.last_frame.get_window_title()) ) ) {
78 /* set icon name and window title */
79 if ( f.get_icon_name() == f.get_window_title() ) {
80 /* write combined Icon Name and Window Title */
81 frame.append( "\033]0;" );
82 const title_type &window_title( f.get_window_title() );
83 for ( title_type::const_iterator i = window_title.begin();
84 i != window_title.end();
85 i++ ) {
86 frame.append( *i );
87 }
88 frame.append( '\007' );
89 /* ST is more correct, but BEL more widely supported */
90 } else {
91 /* write Icon Name */
92 frame.append( "\033]1;" );
93 const title_type &icon_name( f.get_icon_name() );
94 for ( title_type::const_iterator i = icon_name.begin();
95 i != icon_name.end();
96 i++ ) {
97 frame.append( *i );
98 }
99 frame.append( '\007' );
100
101 frame.append( "\033]2;" );
102 const title_type &window_title( f.get_window_title() );
103 for ( title_type::const_iterator i = window_title.begin();
104 i != window_title.end();
105 i++ ) {
106 frame.append( *i );
107 }
108 frame.append( '\007' );
109 }
110
111 }
112
113 /* has clipboard changed? */
114 if (f.get_clipboard() != frame.last_frame.get_clipboard()) {
115 frame.append( "\033]52;c;" );
116 const title_type &clipboard( f.get_clipboard() );
117 for ( title_type::const_iterator i = clipboard.begin();
118 i != clipboard.end();
119 i++ ) {
120 frame.append( *i );
121 }
122 frame.append( '\007' );
123 }
124
125 /* has reverse video state changed? */
126 if ( (!initialized)
127 || (f.ds.reverse_video != frame.last_frame.ds.reverse_video) ) {
128 /* set reverse video */
129 snprintf( tmp, 64, "\033[?5%c", (f.ds.reverse_video ? 'h' : 'l') );
130 frame.append( tmp );
131 }
132
133 /* has size changed? */
134 if ( (!initialized)
135 || (f.ds.get_width() != frame.last_frame.ds.get_width())
136 || (f.ds.get_height() != frame.last_frame.ds.get_height()) ) {
137 /* reset scrolling region */
138 frame.append( "\033[r" );
139
140 /* clear screen */
141 frame.append( "\033[0m\033[H\033[2J" );
142 initialized = false;
143 frame.cursor_x = frame.cursor_y = 0;
144 frame.current_rendition = initial_rendition();
145 } else {
146 frame.cursor_x = frame.last_frame.ds.get_cursor_col();
147 frame.cursor_y = frame.last_frame.ds.get_cursor_row();
148 frame.current_rendition = frame.last_frame.ds.get_renditions();
149 }
150
151 /* is cursor visibility initialized? */
152 if ( !initialized ) {
153 frame.cursor_visible = false;
154 frame.append( "\033[?25l" );
155 }
156
157 int frame_y = 0;
158 Framebuffer::row_pointer blank_row;
159 Framebuffer::rows_type rows( frame.last_frame.get_rows() );
160 /* Extend rows if we've gotten a resize and new is wider than old */
161 if ( frame.last_frame.ds.get_width() < f.ds.get_width() ) {
162 for ( Framebuffer::rows_type::iterator p = rows.begin(); p != rows.end(); p++ ) {
163 *p = make_shared<Row>( **p );
164 (*p)->cells.resize( f.ds.get_width(), Cell( f.ds.get_background_rendition() ) );
165 }
166 }
167 /* Add rows if we've gotten a resize and new is taller than old */
168 if ( static_cast<int>( rows.size() ) < f.ds.get_height() ) {
169 // get a proper blank row
170 const size_t w = f.ds.get_width();
171 const color_type c = 0;
172 blank_row = make_shared<Row>( w, c );
173 rows.resize( f.ds.get_height(), blank_row );
174 }
175
176 /* shortcut -- has display moved up by a certain number of lines? */
177 if ( initialized ) {
178 int lines_scrolled = 0;
179 int scroll_height = 0;
180
181 for ( int row = 0; row < f.ds.get_height(); row++ ) {
182 const Row *new_row = f.get_row( 0 );
183 const Row *old_row = &*rows.at( row );
184 if ( ! ( new_row == old_row || *new_row == *old_row ) ) {
185 continue;
186 }
187 /* if row 0, we're looking at ourselves and probably didn't scroll */
188 if ( row == 0 ) {
189 break;
190 }
191 /* found a scroll */
192 lines_scrolled = row;
193 scroll_height = 1;
194
195 /* how big is the region that was scrolled? */
196 for ( int region_height = 1;
197 lines_scrolled + region_height < f.ds.get_height();
198 region_height++ ) {
199 if ( *f.get_row( region_height )
200 == *rows.at( lines_scrolled + region_height ) ) {
201 scroll_height = region_height + 1;
202 } else {
203 break;
204 }
205 }
206
207 break;
208 }
209
210 if ( scroll_height ) {
211 frame_y = scroll_height;
212
213 if ( lines_scrolled ) {
214 /* Now we need a proper blank row. */
215 if ( blank_row.get() == NULL ) {
216 const size_t w = f.ds.get_width();
217 const color_type c = 0;
218 blank_row = make_shared<Row>( w, c );
219 }
220 frame.update_rendition( initial_rendition(), true );
221
222 int top_margin = 0;
223 int bottom_margin = top_margin + lines_scrolled + scroll_height - 1;
224
225 assert( bottom_margin < f.ds.get_height() );
226
227 /* Common case: if we're already on the bottom line and we're scrolling the whole
228 * screen, just do a CR and LFs.
229 */
230 if ( scroll_height + lines_scrolled == f.ds.get_height()
231 && frame.cursor_y + 1 == f.ds.get_height() ) {
232 frame.append( '\r' );
233 frame.append( lines_scrolled, '\n' );
234 frame.cursor_x = 0;
235 } else {
236 /* set scrolling region */
237 snprintf( tmp, 64, "\033[%d;%dr",
238 top_margin + 1, bottom_margin + 1);
239 frame.append( tmp );
240
241 /* go to bottom of scrolling region */
242 frame.cursor_x = frame.cursor_y = -1;
243 frame.append_silent_move( bottom_margin, 0 );
244
245 /* scroll */
246 frame.append( lines_scrolled, '\n' );
247
248 /* reset scrolling region */
249 frame.append( "\033[r" );
250 /* invalidate cursor position after unsetting scrolling region */
251 frame.cursor_x = frame.cursor_y = -1;
252 }
253
254 /* do the move in our local index */
255 for ( int i = top_margin; i <= bottom_margin; i++ ) {
256 if ( i + lines_scrolled <= bottom_margin ) {
257 rows.at( i ) = rows.at( i + lines_scrolled );
258 } else {
259 rows.at( i ) = blank_row;
260 }
261 }
262 }
263 }
264 }
265
266 /* Now update the display, row by row */
267 bool wrap = false;
268 for ( ; frame_y < f.ds.get_height(); frame_y++ ) {
269 wrap = put_row( initialized, frame, f, frame_y, *rows.at( frame_y ), wrap );
270 }
271
272 /* has cursor location changed? */
273 if ( (!initialized)
274 || (f.ds.get_cursor_row() != frame.cursor_y)
275 || (f.ds.get_cursor_col() != frame.cursor_x) ) {
276 frame.append_move( f.ds.get_cursor_row(), f.ds.get_cursor_col() );
277 }
278
279 /* has cursor visibility changed? */
280 if ( (!initialized)
281 || (f.ds.cursor_visible != frame.cursor_visible) ) {
282 if ( f.ds.cursor_visible ) {
283 frame.append( "\033[?25h" );
284 } else {
285 frame.append( "\033[?25l" );
286 }
287 }
288
289 /* have renditions changed? */
290 frame.update_rendition( f.ds.get_renditions(), !initialized );
291
292 /* has bracketed paste mode changed? */
293 if ( (!initialized)
294 || (f.ds.bracketed_paste != frame.last_frame.ds.bracketed_paste) ) {
295 frame.append( f.ds.bracketed_paste ? "\033[?2004h" : "\033[?2004l" );
296 }
297
298 /* has mouse reporting mode changed? */
299 if ( (!initialized)
300 || (f.ds.mouse_reporting_mode != frame.last_frame.ds.mouse_reporting_mode) ) {
301 if (f.ds.mouse_reporting_mode == DrawState::MOUSE_REPORTING_NONE) {
302 frame.append("\033[?1003l");
303 frame.append("\033[?1002l");
304 frame.append("\033[?1001l");
305 frame.append("\033[?1000l");
306 } else {
307 if (frame.last_frame.ds.mouse_reporting_mode != DrawState::MOUSE_REPORTING_NONE) {
308 snprintf(tmp, sizeof(tmp), "\033[?%dl", frame.last_frame.ds.mouse_reporting_mode);
309 frame.append(tmp);
310 }
311 snprintf(tmp, sizeof(tmp), "\033[?%dh", f.ds.mouse_reporting_mode);
312 frame.append(tmp);
313 }
314 }
315
316 /* has mouse focus mode changed? */
317 if ( (!initialized)
318 || (f.ds.mouse_focus_event != frame.last_frame.ds.mouse_focus_event) ) {
319 frame.append( f.ds.mouse_focus_event ? "\033[?1004h" : "\033[?1004l" );
320 }
321
322 /* has mouse encoding mode changed? */
323 if ( (!initialized)
324 || (f.ds.mouse_encoding_mode != frame.last_frame.ds.mouse_encoding_mode) ) {
325 if (f.ds.mouse_encoding_mode == DrawState::MOUSE_ENCODING_DEFAULT) {
326 frame.append("\033[?1015l");
327 frame.append("\033[?1006l");
328 frame.append("\033[?1005l");
329 } else {
330 if (frame.last_frame.ds.mouse_encoding_mode != DrawState::MOUSE_ENCODING_DEFAULT) {
331 snprintf(tmp, sizeof(tmp), "\033[?%dl", frame.last_frame.ds.mouse_encoding_mode);
332 frame.append(tmp);
333 }
334 snprintf(tmp, sizeof(tmp), "\033[?%dh", f.ds.mouse_encoding_mode);
335 frame.append(tmp);
336 }
337 }
338
339 return frame.str;
340}

342bool Display::put_row( bool initialized, FrameState &frame, const Framebuffer &f, int frame_y, const Row &old_row, bool wrap ) const
343{
344 char tmp[ 64 ];
345 int frame_x = 0;
346
347 const Row &row = *f.get_row( frame_y );
348 const Row::cells_type &cells = row.cells;
349 const Row::cells_type &old_cells = old_row.cells;
350
351 /* If we're forced to write the first column because of wrap, go ahead and do so. */
352 if ( wrap ) {
353 const Cell &cell = cells.at( 0 );
354 frame.update_rendition( cell.get_renditions() );
355 frame.append_cell( cell );
356 frame_x += cell.get_width();
357 frame.cursor_x += cell.get_width();
358 }
359
360 /* If rows are the same object, we don't need to do anything at all. */
361 if (initialized && &row == &old_row ) {
362 return false;
363 }
364
365 const bool wrap_this = row.get_wrap();
366 const int row_width = f.ds.get_width();
367 int clear_count = 0;
368 bool wrote_last_cell = false;
369 Renditions blank_renditions = initial_rendition();
370
371 /* iterate for every cell */
372 while ( frame_x < row_width ) {
373
374 const Cell &cell = cells.at( frame_x );
375
376 /* Does cell need to be drawn? Skip all this. */
377 if ( initialized
378 && !clear_count
379 && ( cell == old_cells.at( frame_x ) ) ) {
380 frame_x += cell.get_width();
381 continue;
382 }
383
384 /* Slurp up all the empty cells */
385 if ( cell.empty() ) {
386 if ( !clear_count ) {
387 blank_renditions = cell.get_renditions();
388 }
389 if ( cell.get_renditions() == blank_renditions ) {
390 /* Remember run of blank cells */
391 clear_count++;
392 frame_x++;
393 continue;
394 }
395 }
396
397 /* Clear or write cells within the row (not to end). */
398 if ( clear_count ) {
399 /* Move to the right position. */
400 frame.append_silent_move( frame_y, frame_x - clear_count );
401 frame.update_rendition( blank_renditions );
402 bool can_use_erase = has_bce || ( frame.current_rendition == initial_rendition() );
403 if ( can_use_erase && has_ech && clear_count > 4 ) {
404 snprintf( tmp, 64, "\033[%dX", clear_count );
405 frame.append( tmp );
406 } else {
407 frame.append( clear_count, ' ' );
408 frame.cursor_x = frame_x;
409 }
410 clear_count = 0;
411 // If the current character is *another* empty cell in a different rendition,
412 // we restart counting and continue here
413 if ( cell.empty() ) {
414 blank_renditions = cell.get_renditions();
415 clear_count = 1;
416 frame_x++;
417 continue;
418 }
419 }
420
421
422 /* Now draw a character cell. */
423 /* Move to the right position. */
424 const int cell_width = cell.get_width();
425 /* If we are about to print the last character in a wrapping row,
426 trash the cursor position to force explicit positioning. We do
427 this because our input terminal state may have the cursor on
428 the autowrap column ("column 81"), but our output terminal
429 states always snap the cursor to the true last column ("column
430 80"), and we want to be able to apply the diff to either, for
431 verification. */
432 if ( wrap_this && frame_x + cell_width >= row_width ) {
433 frame.cursor_x = frame.cursor_y = -1;
434 }
435 frame.append_silent_move( frame_y, frame_x );
436 frame.update_rendition( cell.get_renditions() );
437 frame.append_cell( cell );
438 frame_x += cell_width;
439 frame.cursor_x += cell_width;
440 if ( frame_x >= row_width ) {
441 wrote_last_cell = true;
442 }
443 }
444
445 /* End of line. */
446
447 /* Clear or write empty cells at EOL. */
448 if ( clear_count ) {
449 /* Move to the right position. */
450 frame.append_silent_move( frame_y, frame_x - clear_count );
451 frame.update_rendition( blank_renditions );
452
453 bool can_use_erase = has_bce || ( frame.current_rendition == initial_rendition() );
454 if ( can_use_erase && !wrap_this ) {
455 frame.append( "\033[K" );
456 } else {
457 frame.append( clear_count, ' ' );
458 frame.cursor_x = frame_x;
459 wrote_last_cell = true;
460 }
461 }
462
463 if ( ! ( wrote_last_cell
464 && (frame_y < f.ds.get_height() - 1) ) ) {
465 return false;
466 }
467 /* To hint that a word-select should group the end of one line
468 with the beginning of the next, we let the real cursor
469 actually wrap around in cases where it wrapped around for us. */
470 if ( wrap_this ) {
471 /* Update our cursor, and ask for wrap on the next row. */
472 frame.cursor_x = 0;
473 frame.cursor_y++;
474 return true;
475 }
476 /* Resort to CR/LF and update our cursor. */
477 frame.append( "\r\n" );
478 frame.cursor_x = 0;
479 frame.cursor_y++;
480 return false;
481}

483FrameState::FrameState( const Framebuffer &s_last )
484 : str(), cursor_x(0), cursor_y(0), current_rendition( 0 ),
485 cursor_visible( s_last.ds.cursor_visible ),
486 last_frame( s_last )
487{
488 /* Preallocate for better performance. Make a guess-- doesn't matter for correctness */
489 str.reserve( last_frame.ds.get_width() * last_frame.ds.get_height() * 4 );
490}

492void FrameState::append_silent_move( int y, int x )
493{
494 if ( cursor_x == x && cursor_y == y ) return;
495 /* turn off cursor if necessary before moving cursor */
496 if ( cursor_visible ) {
497 append( "\033[?25l" );
498 cursor_visible = false;
499 }
500 append_move( y, x );
501}
502
503void FrameState::append_move( int y, int x )
504{
505 const int last_x = cursor_x;
506 const int last_y = cursor_y;
507 cursor_x = x;
508 cursor_y = y;
509 // Only optimize if cursor pos is known
510 if ( last_x != -1 && last_y != -1 ) {
511 // Can we use CR and/or LF? They're cheap and easier to trace.
512 if ( x == 0 && y - last_y >= 0 && y - last_y < 5 ) {
513 if ( last_x != 0 ) {
514 append( '\r' );
515 }
516 append( y - last_y, '\n' );
517 return;
518 }
519 // Backspaces are good too.
520 if ( y == last_y && x - last_x < 0 && x - last_x > -5 ) {
521 append( last_x - x, '\b' );
522 return;
523 }
524 // More optimizations are possible.
525 }
526 char tmp[ 64 ];
527 snprintf( tmp, 64, "\033[%d;%dH", y + 1, x + 1 );
528 append( tmp );
529}

531void FrameState::update_rendition(const Renditions &r, bool force) {
532 if ( force || !(current_rendition == r) ) {
533 /* print renditions */
534 append_string( r.sgr() );
535 current_rendition = r;
536 }
