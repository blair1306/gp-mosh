44static string network_order_string( uint16_t host_order )
45{
46 uint16_t net_int = htobe16( host_order );
47 return string( (char *)&net_int, sizeof( net_int ) );
48}
49
50static string network_order_string( uint64_t host_order )
51{
52 uint64_t net_int = htobe64( host_order );
53 return string( (char *)&net_int, sizeof( net_int ) );
54}
55
56string Fragment::tostring( void )
57{
58 assert( initialized );
59
60 string ret;
61
62 ret += network_order_string( id );
63
64 fatal_assert( !( fragment_num & 0x8000 ) ); /* effective limit on size of a terminal screen change or buffered user input */
65 uint16_t combined_fragment_num = ( final << 15 ) | fragment_num;
66 ret += network_order_string( combined_fragment_num );
67
68 assert( ret.size() == frag_header_len );
69
70 ret += contents;
71
72 return ret;
73}

75Fragment::Fragment( const string &x )
76 : id( -1 ), fragment_num( -1 ), final( false ), initialized( true ),
77 contents()
78{
79 fatal_assert( x.size() >= frag_header_len );
80 contents = string( x.begin() + frag_header_len, x.end() );
81
82 uint64_t data64;
83 uint16_t *data16 = (uint16_t *)x.data();
84 memcpy( &data64, x.data(), sizeof( data64 ) );
85 id = be64toh( data64 );
86 fragment_num = be16toh( data16[ 4 ] );
87 final = ( fragment_num & 0x8000 ) >> 15;
88 fragment_num &= 0x7FFF;
89}

91bool FragmentAssembly::add_fragment( Fragment &frag )
92{
93 /* see if this is a totally new packet */
94 if ( current_id != frag.id ) {
95 fragments.clear();
96 fragments.resize( frag.fragment_num + 1 );
97 fragments.at( frag.fragment_num ) = frag;
98 fragments_arrived = 1;
99 fragments_total = -1; /* unknown */
100 current_id = frag.id;
101 } else { /* not a new packet */
102 /* see if we already have this fragment */
103 if ( (fragments.size() > frag.fragment_num)
104 && (fragments.at( frag.fragment_num ).initialized) ) {
105 /* make sure new version is same as what we already have */
106 assert( fragments.at( frag.fragment_num ) == frag );
107 } else {
108 if ( (int)fragments.size() < frag.fragment_num + 1 ) {
109 fragments.resize( frag.fragment_num + 1 );
110 }
111 fragments.at( frag.fragment_num ) = frag;
112 fragments_arrived++;
113 }
114 }
115
116 if ( frag.final ) {
117 fragments_total = frag.fragment_num + 1;
118 assert( (int)fragments.size() <= fragments_total );
119 fragments.resize( fragments_total );
120 }
121
122 if ( fragments_total != -1 ) {
123 assert( fragments_arrived <= fragments_total );
124 }
125
126 /* see if we're done */
127 return fragments_arrived == fragments_total;
128}
129
130Instruction FragmentAssembly::get_assembly( void )
131{
132 assert( fragments_arrived == fragments_total );
133
134 string encoded;
135
136 for ( int i = 0; i < fragments_total; i++ ) {
137 assert( fragments.at( i ).initialized );
138 encoded += fragments.at( i ).contents;
139 }
140
141 Instruction ret;
142 fatal_assert( ret.ParseFromString( get_compressor().uncompress_str( encoded ) ) );
143
144 fragments.clear();
145 fragments_arrived = 0;
146 fragments_total = -1;
147
148 return ret;
149}

151bool Fragment::operator==( const Fragment &x ) const
152{
153 return ( id == x.id ) && ( fragment_num == x.fragment_num ) && ( final == x.final )
154 && ( initialized == x.initialized ) && ( contents == x.contents );
155}

157vector<Fragment> Fragmenter::make_fragments( const Instruction &inst, size_t MTU )
158{
159 MTU -= Fragment::frag_header_len;
160 if ( (inst.old_num() != last_instruction.old_num())
161 || (inst.new_num() != last_instruction.new_num())
162 || (inst.ack_num() != last_instruction.ack_num())
163 || (inst.throwaway_num() != last_instruction.throwaway_num())
164 || (inst.chaff() != last_instruction.chaff())
165 || (inst.protocol_version() != last_instruction.protocol_version())
166 || (last_MTU != MTU) ) {
167 next_instruction_id++;
168 }
169
170 if ( (inst.old_num() == last_instruction.old_num())
171 && (inst.new_num() == last_instruction.new_num()) ) {
172 assert( inst.diff() == last_instruction.diff() );
173 }
174
175 last_instruction = inst;
176 last_MTU = MTU;
177
178 string payload = get_compressor().compress_str( inst.SerializeAsString() );
179 uint16_t fragment_num = 0;
180 vector<Fragment> ret;
181
182 while ( !payload.empty() ) {
183 string this_fragment;
184 bool final = false;
185
186 if ( payload.size() > MTU ) {
187 this_fragment = string( payload.begin(), payload.begin() + MTU );
188 payload = string( payload.begin() + MTU, payload.end() );
189 } else {
190 this_fragment = payload;
191 payload.clear();
192 final = true;
193 }
194
195 ret.push_back( Fragment( next_instruction_id, fragment_num++, final, this_fragment ) );
196 }
197
198 return ret;
199}
