67const uint64_t DIRECTION_MASK = uint64_t(1) << 63;
68const uint64_t SEQUENCE_MASK = uint64_t(-1) ^ DIRECTION_MASK;
69
70/* Read in packet */
71Packet::Packet( const Message & message )
72 : seq( message.nonce.val() & SEQUENCE_MASK ),
73 direction( (message.nonce.val() & DIRECTION_MASK) ? TO_CLIENT : TO_SERVER ),
74 timestamp( -1 ),
75 timestamp_reply( -1 ),
76 payload()
77{
78 dos_assert( message.text.size() >= 2 * sizeof( uint16_t ) );
79
80 const uint16_t *data = (uint16_t *)message.text.data();
81 timestamp = be16toh( data[ 0 ] );
82 timestamp_reply = be16toh( data[ 1 ] );
83
84 payload = string( message.text.begin() + 2 * sizeof( uint16_t ), message.text.end() );
85}
86
87/* Output from packet */
88Message Packet::toMessage( void )
89{
90 uint64_t direction_seq = (uint64_t( direction == TO_CLIENT ) << 63) | (seq & SEQUENCE_MASK);
91
92 uint16_t ts_net[ 2 ] = { static_cast<uint16_t>( htobe16( timestamp ) ),
93 static_cast<uint16_t>( htobe16( timestamp_reply ) ) };
94
95 string timestamps = string( (char *)ts_net, 2 * sizeof( uint16_t ) );
96
97 return Message( Nonce( direction_seq ), timestamps + payload );
98}

100Packet Connection::new_packet( const string &s_payload )
101{
102 uint16_t outgoing_timestamp_reply = -1;
103
104 uint64_t now = timestamp();
105
106 if ( now - saved_timestamp_received_at < 1000 ) { /* we have a recent received timestamp */
107 /* send "corrected" timestamp advanced by how long we held it */
108 outgoing_timestamp_reply = saved_timestamp + (now - saved_timestamp_received_at);
109 saved_timestamp = -1;
110 saved_timestamp_received_at = 0;
111 }
112
113 Packet p( direction, timestamp16(), outgoing_timestamp_reply, s_payload );
114
115 return p;
116}

129void Connection::prune_sockets( void )
130{
131 /* don't keep old sockets if the new socket has been working for long enough */
132 if ( socks.size() > 1 ) {
133 if ( timestamp() - last_port_choice > MAX_OLD_SOCKET_AGE ) {
134 int num_to_kill = socks.size() - 1;
135 for ( int i = 0; i < num_to_kill; i++ ) {
136 socks.pop_front();
137 }
138 }
139 } else {
140 return;
141 }
142
143 /* make sure we don't have too many receive sockets open */
144 if ( socks.size() > MAX_PORTS_OPEN ) {
145 int num_to_kill = socks.size() - MAX_PORTS_OPEN;
146 for ( int i = 0; i < num_to_kill; i++ ) {
147 socks.pop_front();
148 }
149 }
150}

152Connection::Socket::Socket( int family )
153 : _fd( socket( family, SOCK_DGRAM, 0 ) )
154{
155 if ( _fd < 0 ) {
156 throw NetworkException( "socket", errno );
157 }
158
159 /* Disable path MTU discovery */
160#ifdef HAVE_IP_MTU_DISCOVER
161 int flag = IP_PMTUDISC_DONT;
162 if ( setsockopt( _fd, IPPROTO_IP, IP_MTU_DISCOVER, &flag, sizeof flag ) < 0 ) {
163 throw NetworkException( "setsockopt", errno );
164 }
165#endif
166
167 // int dscp = 0x92; /* OS X does not have IPTOS_DSCP_AF42 constant */
168 int dscp = 0x02; /* ECN-capable transport only */
169 if ( setsockopt( _fd, IPPROTO_IP, IP_TOS, &dscp, sizeof dscp ) < 0 ) {
170 // perror( "setsockopt( IP_TOS )" );
171 }
172
173 /* request explicit congestion notification on received datagrams */
174#ifdef HAVE_IP_RECVTOS
175 int tosflag = true;
176 if ( setsockopt( _fd, IPPROTO_IP, IP_RECVTOS, &tosflag, sizeof tosflag ) < 0
177 && family == IPPROTO_IP ) { /* FreeBSD disallows this option on IPv6 sockets. */
178 perror( "setsockopt( IP_RECVTOS )" );
179 }
180#endif
181}

201void Connection::set_MTU( int family )
202{
203 switch ( family ) {
204 case AF_INET:
205 MTU = DEFAULT_IPV4_MTU - IPV4_HEADER_LEN;
206 break;
207 case AF_INET6:
208 MTU = DEFAULT_IPV6_MTU - IPV6_HEADER_LEN;
209 break;
210 default:
211 throw NetworkException( "Unknown address family", 0 );
212 }
213}

216public:
217 struct addrinfo *res;
218 AddrInfo( const char *node, const char *service,
219 const struct addrinfo *hints ) :
220 res( NULL ) {
221 int errcode = getaddrinfo( node, service, hints, &res );
222 if ( errcode != 0 ) {
223 throw NetworkException( std::string( "Bad IP address (" ) + (node != NULL ? node : "(null)") + "): " + gai_strerror( errcode ), 0 );

232Connection::Connection( const char *desired_ip, const char *desired_port ) /* server */
233 : socks(),
234 has_remote_addr( false ),
235 remote_addr(),
236 remote_addr_len( 0 ),
237 server( true ),
238 MTU( DEFAULT_SEND_MTU ),
239 key(),
240 session( key ),
241 direction( TO_CLIENT ),
242 saved_timestamp( -1 ),
243 saved_timestamp_received_at( 0 ),
244 expected_receiver_seq( 0 ),
245 last_heard( -1 ),
246 last_port_choice( -1 ),
247 last_roundtrip_success( -1 ),
248 RTT_hit( false ),
249 SRTT( 1000 ),
250 RTTVAR( 500 ),
251 send_error()
252{
253 setup();
254
255 /* The mosh wrapper always gives an IP request, in order
256 to deal with multihomed servers. The port is optional. */
257
258 /* If an IP request is given, we try to bind to that IP, but we also
259 try INADDR_ANY. If a port request is given, we bind only to that port. */
260
261 /* convert port numbers */
262 int desired_port_low = -1;
263 int desired_port_high = -1;
264
265 if ( desired_port && !parse_portrange( desired_port, desired_port_low, desired_port_high ) ) {
266 throw NetworkException("Invalid port range", 0);
267 }
268
269 /* try to bind to desired IP first */
270 if ( desired_ip ) {
271 try {
272 if ( try_bind( desired_ip, desired_port_low, desired_port_high ) ) { return; }
273 } catch ( const NetworkException &e ) {
274 fprintf( stderr, "Error binding to IP %s: %s\n",
275 desired_ip,
276 e.what() );
277 }
278 }
279
280 /* now try any local interface */
281 try {
282 if ( try_bind( NULL, desired_port_low, desired_port_high ) ) { return; }
283 } catch ( const NetworkException &e ) {
284 fprintf( stderr, "Error binding to any interface: %s\n",
285 e.what() );
286 throw; /* this time it's fatal */
287 }
288
289 throw NetworkException( "Could not bind", errno );
290}

292bool Connection::try_bind( const char *addr, int port_low, int port_high )
293{
294 struct addrinfo hints;
295 memset( &hints, 0, sizeof( hints ) );
296 hints.ai_family = AF_UNSPEC;
297 hints.ai_socktype = SOCK_DGRAM;
298 hints.ai_flags = AI_PASSIVE | AI_NUMERICHOST | AI_NUMERICSERV;
299 AddrInfo ai( addr, "0", &hints );
300
301 Addr local_addr;
302 socklen_t local_addr_len = ai.res->ai_addrlen;
303 memcpy( &local_addr.sa, ai.res->ai_addr, local_addr_len );
304
305 int search_low = PORT_RANGE_LOW, search_high = PORT_RANGE_HIGH;
306
307 if ( port_low != -1 ) { /* low port preference */
308 search_low = port_low;
309 }
310 if ( port_high != -1 ) { /* high port preference */
311 search_high = port_high;
312 }
313
314 socks.push_back( Socket( local_addr.sa.sa_family ) );
315 for ( int i = search_low; i <= search_high; i++ ) {
316 switch (local_addr.sa.sa_family) {
317 case AF_INET:
318 local_addr.sin.sin_port = htons( i );
319 break;
320 case AF_INET6:
321 local_addr.sin6.sin6_port = htons( i );
322 break;
323 default:
324 throw NetworkException( "Unknown address family", 0 );
325 }
326
327 if ( local_addr.sa.sa_family == AF_INET6
328 && memcmp(&local_addr.sin6.sin6_addr, &in6addr_any, sizeof(in6addr_any)) == 0 ) {
329 const int off = 0;
330 if ( setsockopt( sock(), IPPROTO_IPV6, IPV6_V6ONLY, &off, sizeof(off) ) ) {
331 perror( "setsockopt( IPV6_V6ONLY, off )" );
332 }
333 }
334
335 if ( bind( sock(), &local_addr.sa, local_addr_len ) == 0 ) {
336 set_MTU( local_addr.sa.sa_family );
337 return true;
338 } // else fallthrough to below code, on last iteration.
339 }
340 int saved_errno = errno;
341 socks.pop_back();
342 char host[ NI_MAXHOST ], serv[ NI_MAXSERV ];
343 int errcode = getnameinfo( &local_addr.sa, local_addr_len,
344 host, sizeof( host ), serv, sizeof( serv ),
345 NI_DGRAM | NI_NUMERICHOST | NI_NUMERICSERV );
346 if ( errcode != 0 ) {
347 throw NetworkException( std::string( "bind: getnameinfo: " ) + gai_strerror( errcode ), 0 );
348 }
349 fprintf( stderr, "Failed binding to %s:%s\n",
350 host, serv );
351 throw NetworkException( "bind", saved_errno );
352}

354Connection::Connection( const char *key_str, const char *ip, const char *port ) /* client */
355 : socks(),
356 has_remote_addr( false ),
357 remote_addr(),
358 remote_addr_len( 0 ),
359 server( false ),
360 MTU( DEFAULT_SEND_MTU ),
361 key( key_str ),
362 session( key ),
363 direction( TO_SERVER ),
364 saved_timestamp( -1 ),
365 saved_timestamp_received_at( 0 ),
366 expected_receiver_seq( 0 ),
367 last_heard( -1 ),
368 last_port_choice( -1 ),
369 last_roundtrip_success( -1 ),
370 RTT_hit( false ),
371 SRTT( 1000 ),
372 RTTVAR( 500 ),
373 send_error()
374{
375 setup();
376
377 /* associate socket with remote host and port */
378 struct addrinfo hints;
379 memset( &hints, 0, sizeof( hints ) );
380 hints.ai_family = AF_UNSPEC;
381 hints.ai_socktype = SOCK_DGRAM;
382 hints.ai_flags = AI_NUMERICHOST | AI_NUMERICSERV;
383 AddrInfo ai( ip, port, &hints );
384 fatal_assert( static_cast<size_t>( ai.res->ai_addrlen ) <= sizeof( remote_addr ) );
385 remote_addr_len = ai.res->ai_addrlen;
386 memcpy( &remote_addr.sa, ai.res->ai_addr, remote_addr_len );
387
388 has_remote_addr = true;
389
390 socks.push_back( Socket( remote_addr.sa.sa_family ) );
391
392 set_MTU( remote_addr.sa.sa_family );
393}

395void Connection::send( const string & s )
396{
397 if ( !has_remote_addr ) {
398 return;
399 }
400
401 Packet px = new_packet( s );
402
403 string p = session.encrypt( px.toMessage() );
404
405 ssize_t bytes_sent = sendto( sock(), p.data(), p.size(), MSG_DONTWAIT,
406 &remote_addr.sa, remote_addr_len );
407
408 if ( bytes_sent != static_cast<ssize_t>( p.size() ) ) {
409 /* Make sendto() failure available to the frontend. */
410 send_error = "sendto: ";
411 send_error += strerror( errno );
412
413 if ( errno == EMSGSIZE ) {
414 MTU = DEFAULT_SEND_MTU; /* payload MTU of last resort */
415 }
416 }
417
418 uint64_t now = timestamp();
419 if ( server ) {
420 if ( now - last_heard > SERVER_ASSOCIATION_TIMEOUT ) {
421 has_remote_addr = false;
422 fprintf( stderr, "Server now detached from client.\n" );
423 }
424 } else { /* client */
425 if ( ( now - last_port_choice > PORT_HOP_INTERVAL )
426 && ( now - last_roundtrip_success > PORT_HOP_INTERVAL ) ) {
427 hop_port();
428 }
429 }
430}

432string Connection::recv( void )
433{
434 assert( !socks.empty() );
435 for ( std::deque< Socket >::const_iterator it = socks.begin();
436 it != socks.end();
437 it++ ) {
438 string payload;
439 try {
440 payload = recv_one( it->fd());
441 } catch ( NetworkException & e ) {
442 if ( (e.the_errno == EAGAIN)
443 || (e.the_errno == EWOULDBLOCK) ) {
444 continue;
445 } else {
446 throw;
447 }
448 }
449
450 /* succeeded */
451 prune_sockets();
452 return payload;
453 }
454 throw NetworkException( "No packet received" );
455}

457string Connection::recv_one( int sock_to_recv )
458{
459 /* receive source address, ECN, and payload in msghdr structure */
460 Addr packet_remote_addr;
461 struct msghdr header;
462 struct iovec msg_iovec;
463
464 char msg_payload[ Session::RECEIVE_MTU ];
465 char msg_control[ Session::RECEIVE_MTU ];
466
467 /* receive source address */
468 header.msg_name = &packet_remote_addr;
469 header.msg_namelen = sizeof packet_remote_addr;
470
471 /* receive payload */
472 msg_iovec.iov_base = msg_payload;
473 msg_iovec.iov_len = sizeof msg_payload;
474 header.msg_iov = &msg_iovec;
475 header.msg_iovlen = 1;
476
477 /* receive explicit congestion notification */
478 header.msg_control = msg_control;
479 header.msg_controllen = sizeof msg_control;
480
481 /* receive flags */
482 header.msg_flags = 0;
483
484 ssize_t received_len = recvmsg( sock_to_recv, &header, MSG_DONTWAIT );
485
486 if ( received_len < 0 ) {
487 throw NetworkException( "recvmsg", errno );
488 }
489
490 if ( header.msg_flags & MSG_TRUNC ) {
491 throw NetworkException( "Received oversize datagram", errno );
492 }
493
494 /* receive ECN */
495 bool congestion_experienced = false;
496
497 struct cmsghdr *ecn_hdr = CMSG_FIRSTHDR( &header );
498 if ( ecn_hdr
499 && ecn_hdr->cmsg_level == IPPROTO_IP
500 && ( ecn_hdr->cmsg_type == IP_TOS
501#ifdef IP_RECVTOS
502 || ecn_hdr->cmsg_type == IP_RECVTOS
503#endif
504 ) ) {
505 /* got one */
506 uint8_t *ecn_octet_p = (uint8_t *)CMSG_DATA( ecn_hdr );
507 assert( ecn_octet_p );
508
509 congestion_experienced = (*ecn_octet_p & 0x03) == 0x03;
510 }
511
512 Packet p( session.decrypt( msg_payload, received_len ) );
513
514 dos_assert( p.direction == (server ? TO_SERVER : TO_CLIENT) ); /* prevent malicious playback to sender */
515
516 if ( p.seq < expected_receiver_seq ) { /* don't use (but do return) out-of-order packets for timestamp or targeting */
517 return p.payload;
518 }
519 expected_receiver_seq = p.seq + 1; /* this is security-sensitive because a replay attack could otherwise
520 screw up the timestamp and targeting */
521
522 if ( p.timestamp != uint16_t(-1) ) {
523 saved_timestamp = p.timestamp;
524 saved_timestamp_received_at = timestamp();
525
526 if ( congestion_experienced ) {
527 /* signal counterparty to slow down */
528 /* this will gradually slow the counterparty down to the minimum frame rate */
529 saved_timestamp -= CONGESTION_TIMESTAMP_PENALTY;
530 if ( server ) {
531 fprintf( stderr, "Received explicit congestion notification.\n" );
532 }
533 }
534 }
535
536 if ( p.timestamp_reply != uint16_t(-1) ) {
537 uint16_t now = timestamp16();
538 double R = timestamp_diff( now, p.timestamp_reply );
539
540 if ( R < 5000 ) { /* ignore large values, e.g. server was Ctrl-Zed */
541 if ( !RTT_hit ) { /* first measurement */
542 SRTT = R;
543 RTTVAR = R / 2;
544 RTT_hit = true;
545 } else {
546 const double alpha = 1.0 / 8.0;
547 const double beta = 1.0 / 4.0;
548
549 RTTVAR = (1 - beta) * RTTVAR + ( beta * fabs( SRTT - R ) );
550 SRTT = (1 - alpha) * SRTT + ( alpha * R );
551 }
552 }
553 }
554
555 /* auto-adjust to remote host */
556 has_remote_addr = true;
557 last_heard = timestamp();
558
559 if ( server && /* only client can roam */
560 ( remote_addr_len != header.msg_namelen ||
561 memcmp( &remote_addr, &packet_remote_addr, remote_addr_len ) != 0 ) ) {
562 remote_addr = packet_remote_addr;
563 remote_addr_len = header.msg_namelen;
564 char host[ NI_MAXHOST ], serv[ NI_MAXSERV ];
565 int errcode = getnameinfo( &remote_addr.sa, remote_addr_len,
566 host, sizeof( host ), serv, sizeof( serv ),
567 NI_DGRAM | NI_NUMERICHOST | NI_NUMERICSERV );
568 if ( errcode != 0 ) {
569 throw NetworkException( std::string( "recv_one: getnameinfo: " ) + gai_strerror( errcode ), 0 );
570 }
571 fprintf( stderr, "Server now attached to client at %s:%s\n",
572 host, serv );
573 }
574 return p.payload;
575}

577std::string Connection::port( void ) const
578{
579 Addr local_addr;
580 socklen_t addrlen = sizeof( local_addr );
581
582 if ( getsockname( sock(), &local_addr.sa, &addrlen ) < 0 ) {
583 throw NetworkException( "getsockname", errno );
584 }
585
586 char serv[ NI_MAXSERV ];
587 int errcode = getnameinfo( &local_addr.sa, addrlen,
588 NULL, 0, serv, sizeof( serv ),
589 NI_DGRAM | NI_NUMERICSERV );
590 if ( errcode != 0 ) {
591 throw NetworkException( std::string( "port: getnameinfo: " ) + gai_strerror( errcode ), 0 );
592 }
593
594 return std::string( serv );
595}

604 uint16_t ts = timestamp() % 65536;

611uint16_t Network::timestamp_diff( uint16_t tsnew, uint16_t tsold )
612{
613 int diff = tsnew - tsold;
614 if ( diff < 0 ) {
615 diff += 65536;
616 }
617
618 assert( diff >= 0 );
619 assert( diff <= 65535 );
620
621 return diff;
622}

624uint64_t Connection::timeout( void ) const
625{
626 uint64_t RTO = lrint( ceil( SRTT + 4 * RTTVAR ) );

628 RTO = MIN_RTO;
629 } else if ( RTO > MAX_RTO ) {
630 RTO = MAX_RTO;
631 }
632 return RTO;

657bool Connection::parse_portrange( const char * desired_port, int & desired_port_low, int & desired_port_high )
658{
659 /* parse "port" or "portlow:porthigh" */
660 desired_port_low = desired_port_high = 0;
661 char *end;
662 long value;
663
664 /* parse first (only?) port */
665 errno = 0;
666 value = strtol( desired_port, &end, 10 );
667 if ( (errno != 0) || (*end != '\0' && *end != ':') ) {
668 fprintf( stderr, "Invalid (low) port number (%s)\n", desired_port );
669 return false;
670 }
671 if ( (value < 0) || (value > 65535) ) {
672 fprintf( stderr, "(Low) port number %ld outside valid range [0..65535]\n", value );
673 return false;
674 }
675
676 desired_port_low = (int)value;
677 if (*end == '\0') { /* not a port range */
678 desired_port_high = desired_port_low;
679 return true;
680 }
681 /* port range; parse high port */
682 const char * cp = end + 1;
683 errno = 0;
684 value = strtol( cp, &end, 10 );
685 if ( (errno != 0) || (*end != '\0') ) {
686 fprintf( stderr, "Invalid high port number (%s)\n", cp );
687 return false;
688 }
689 if ( (value < 0) || (value > 65535) ) {
690 fprintf( stderr, "High port number %ld outside valid range [0..65535]\n", value );
691 return false;
692 }
693
694 desired_port_high = (int)value;
695 if ( desired_port_low > desired_port_high ) {
696 fprintf( stderr, "Low port %d greater than high port %d\n", desired_port_low, desired_port_high );
697 return false;
698 }
699
700 if ( desired_port_low == 0 ) {
701 fprintf( stderr, "Low port 0 incompatible with port ranges\n" );
702 return false;
703 }
704
705
706 return true;
707}
