71template <class MyState, class RemoteState>
72void Transport<MyState, RemoteState>::recv( void )
73{
74 string s( connection.recv() );
75 Fragment frag( s );
76
77 if ( fragments.add_fragment( frag ) ) { /* complete packet */
78 Instruction inst = fragments.get_assembly();
79
80 if ( inst.protocol_version() != MOSH_PROTOCOL_VERSION ) {
81 throw NetworkException( "mosh protocol version mismatch", 0 );
82 }
83
84 sender.process_acknowledgment_through( inst.ack_num() );
85
86 /* inform network layer of roundtrip (end-to-end-to-end) connectivity */
87 connection.set_last_roundtrip_success( sender.get_sent_state_acked_timestamp() );
88
89 /* first, make sure we don't already have the new state */
90 for ( typename list< TimestampedState<RemoteState> >::iterator i = received_states.begin();
91 i != received_states.end();
92 i++ ) {
93 if ( inst.new_num() == i->num ) {
94 return;
95 }
96 }
97
98 /* now, make sure we do have the old state */
99 bool found = 0;
100 typename list< TimestampedState<RemoteState> >::iterator reference_state = received_states.begin();
101 while ( reference_state != received_states.end() ) {
102 if ( inst.old_num() == reference_state->num ) {
103 found = true;
104 break;
105 }
106 reference_state++;
107 }
108
109 if ( !found ) {
110 // fprintf( stderr, "Ignoring out-of-order packet. Reference state %d has been discarded or hasn't yet been received.\n", int(inst.old_num) );
111 return; /* this is security-sensitive and part of how we enforce idempotency */
112 }
113
114 /* Do not accept state if our queue is full */
115 /* This is better than dropping states from the middle of the
116 queue (as sender does), because we don't want to ACK a state
117 and then discard it later. */
118
119 process_throwaway_until( inst.throwaway_num() );
120
121 if ( received_states.size() > 1024 ) { /* limit on state queue */
122 uint64_t now = timestamp();
123 if ( now < receiver_quench_timer ) { /* deny letting state grow further */
124 if ( verbose ) {
125 fprintf( stderr, "[%u] Receiver queue full, discarding %d (malicious sender or long-unidirectional connectivity?)\n",
126 (unsigned int)(timestamp() % 100000), (int)inst.new_num() );
127 }
128 return;
129 } else {
130 receiver_quench_timer = now + 15000;
131 }
132 }
133
134 /* apply diff to reference state */
135 TimestampedState<RemoteState> new_state = *reference_state;
136 new_state.timestamp = timestamp();
137 new_state.num = inst.new_num();
138
139 if ( !inst.diff().empty() ) {
140 new_state.state.apply_string( inst.diff() );
141 }
142
143 /* Insert new state in sorted place */
144 for ( typename list< TimestampedState<RemoteState> >::iterator i = received_states.begin();
145 i != received_states.end();
146 i++ ) {
147 if ( i->num > new_state.num ) {
148 received_states.insert( i, new_state );
149 if ( verbose ) {
150 fprintf( stderr, "[%u] Received OUT-OF-ORDER state %d [ack %d]\n",
151 (unsigned int)(timestamp() % 100000), (int)new_state.num, (int)inst.ack_num() );
152 }
153 return;
154 }
155 }
156 if ( verbose ) {
157 fprintf( stderr, "[%u] Received state %d [coming from %d, ack %d]\n",
158 (unsigned int)(timestamp() % 100000), (int)new_state.num, (int)inst.old_num(), (int)inst.ack_num() );
159 }
160 received_states.push_back( new_state );
161 sender.set_ack_num( received_states.back().num );
162
163 sender.remote_heard( new_state.timestamp );
164 if ( !inst.diff().empty() ) {
165 sender.set_data_ack();
166 }
167 }
168}

170/* The sender uses throwaway_num to tell us the earliest received state that we need to keep around */
171template <class MyState, class RemoteState>
172void Transport<MyState, RemoteState>::process_throwaway_until( uint64_t throwaway_num )
173{
174 typename list< TimestampedState<RemoteState> >::iterator i = received_states.begin();
175 while ( i != received_states.end() ) {
176 typename list< TimestampedState<RemoteState> >::iterator inext = i;
177 inext++;
178 if ( i->num < throwaway_num ) {
179 received_states.erase( i );
180 }
181 i = inext;
182 }
183
184 fatal_assert( received_states.size() > 0 );
185}

187template <class MyState, class RemoteState>
188string Transport<MyState, RemoteState>::get_remote_diff( void )
189{
190 /* find diff between last receiver state and current remote state, then rationalize states */
191
192 string ret( received_states.back().state.diff_from( last_receiver_state ) );
193
194 const RemoteState *oldest_receiver_state = &received_states.front().state;
195
196 for ( typename list< TimestampedState<RemoteState> >::reverse_iterator i = received_states.rbegin();
197 i != received_states.rend();
198 i++ ) {
199 i->state.subtract( oldest_receiver_state );
200 }
201
202 last_receiver_state = received_states.back().state;
203
204 return ret;
205}
