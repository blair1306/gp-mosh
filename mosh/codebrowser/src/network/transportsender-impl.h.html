50template <class MyState>
51TransportSender<MyState>::TransportSender( Connection *s_connection, MyState &initial_state )
52 : connection( s_connection ),
53 current_state( initial_state ),
54 sent_states( 1, TimestampedState<MyState>( timestamp(), 0, initial_state ) ),
55 assumed_receiver_state( sent_states.begin() ),
56 fragmenter(),
57 next_ack_time( timestamp() ),
58 next_send_time( timestamp() ),
59 verbose( 0 ),
60 shutdown_in_progress( false ),
61 shutdown_tries( 0 ),
62 shutdown_start( -1 ),
63 ack_num( 0 ),
64 pending_data_ack( false ),
65 SEND_MINDELAY( 8 ),
66 last_heard( 0 ),
67 prng(),
68 mindelay_clock( -1 )
69{
70}

72/* Try to send roughly two frames per RTT, bounded by limits on frame rate */
73template <class MyState>
74unsigned int TransportSender<MyState>::send_interval( void ) const
75{
76 int SEND_INTERVAL = lrint( ceil( connection->get_SRTT() / 2.0 ) );
77 if ( SEND_INTERVAL < SEND_INTERVAL_MIN ) {
78 SEND_INTERVAL = SEND_INTERVAL_MIN;
79 } else if ( SEND_INTERVAL > SEND_INTERVAL_MAX ) {
80 SEND_INTERVAL = SEND_INTERVAL_MAX;
81 }
82
83 return SEND_INTERVAL;
84}

86/* Housekeeping routine to calculate next send and ack times */
87template <class MyState>
88void TransportSender<MyState>::calculate_timers( void )
89{
90 uint64_t now = timestamp();
91
92 /* Update assumed receiver state */
93 update_assumed_receiver_state();
94
95 /* Cut out common prefix of all states */
96 rationalize_states();
97
98 if ( pending_data_ack && (next_ack_time > now + ACK_DELAY) ) {
99 next_ack_time = now + ACK_DELAY;
100 }
101
102 if ( !(current_state == sent_states.back().state) ) {
103 if ( mindelay_clock == uint64_t( -1 ) ) {
104 mindelay_clock = now;
105 }
106
107 next_send_time = max( mindelay_clock + SEND_MINDELAY,
108 sent_states.back().timestamp + send_interval() );
109 } else if ( !(current_state == assumed_receiver_state->state)
110 && (last_heard + ACTIVE_RETRY_TIMEOUT > now) ) {
111 next_send_time = sent_states.back().timestamp + send_interval();
112 if ( mindelay_clock != uint64_t( -1 ) ) {
113 next_send_time = max( next_send_time, mindelay_clock + SEND_MINDELAY );
114 }
115 } else if ( !(current_state == sent_states.front().state )
116 && (last_heard + ACTIVE_RETRY_TIMEOUT > now) ) {
117 next_send_time = sent_states.back().timestamp + connection->timeout() + ACK_DELAY;
118 } else {
119 next_send_time = uint64_t(-1);
120 }
121
122 /* speed up shutdown sequence */
123 if ( shutdown_in_progress || (ack_num == uint64_t(-1)) ) {
124 next_ack_time = sent_states.back().timestamp + send_interval();
125 }
126}

128/* How many ms to wait until next event */
129template <class MyState>
130int TransportSender<MyState>::wait_time( void )
131{
132 calculate_timers();
133
134 uint64_t next_wakeup = next_ack_time;
135 if ( next_send_time < next_wakeup ) {
136 next_wakeup = next_send_time;
137 }
138
139 uint64_t now = timestamp();
140
141 if ( !connection->get_has_remote_addr() ) {
142 return INT_MAX;
143 }
144
145 if ( next_wakeup > now ) {
146 return next_wakeup - now;
147 } else {
148 return 0;
149 }
150}

152/* Send data or an empty ack if necessary */
153template <class MyState>
154void TransportSender<MyState>::tick( void )
155{
156 calculate_timers(); /* updates assumed receiver state and rationalizes */
157
158 if ( !connection->get_has_remote_addr() ) {
159 return;
160 }
161
162 uint64_t now = timestamp();
163
164 if ( (now < next_ack_time)
165 && (now < next_send_time) ) {
166 return;
167 }
168
169 /* Determine if a new diff or empty ack needs to be sent */
170
171 string diff = current_state.diff_from( assumed_receiver_state->state );
172
173 attempt_prospective_resend_optimization( diff );
174
175 if ( verbose ) {
176 /* verify diff has round-trip identity (modulo Unicode fallback rendering) */
177 MyState newstate( assumed_receiver_state->state );
178 newstate.apply_string( diff );
179 if ( current_state.compare( newstate ) ) {
180 fprintf( stderr, "Warning, round-trip Instruction verification failed!\n" );
181 }
182 /* Also verify that both the original frame and generated frame have the same initial diff. */
183 std::string current_diff( current_state.init_diff() );
184 std::string new_diff( newstate.init_diff() );
185 if ( current_diff != new_diff ) {
186 fprintf( stderr, "Warning, target state Instruction verification failed!\n" );
187 }
188 }
189
190 if ( diff.empty() ) {
191 if ( (now >= next_ack_time) ) {
192 send_empty_ack();
193 mindelay_clock = uint64_t( -1 );
194 }
195 if ( (now >= next_send_time) ) {
196 next_send_time = uint64_t( -1 );
197 mindelay_clock = uint64_t( -1 );
198 }
199 } else if ( (now >= next_send_time) || (now >= next_ack_time) ) {
200 /* Send diffs or ack */
201 send_to_receiver( diff );
202 mindelay_clock = uint64_t( -1 );
203 }
204}

206template <class MyState>
207void TransportSender<MyState>::send_empty_ack( void )
208{
209 uint64_t now = timestamp();
210
211 assert( now >= next_ack_time );
212
213 uint64_t new_num = sent_states.back().num + 1;
214
215 /* special case for shutdown sequence */
216 if ( shutdown_in_progress ) {
217 new_num = uint64_t( -1 );
218 }
219
220 // sent_states.push_back( TimestampedState<MyState>( sent_states.back().timestamp, new_num, current_state ) );
221 add_sent_state( now, new_num, current_state );
222 send_in_fragments( "", new_num );
223
224 next_ack_time = now + ACK_INTERVAL;
225 next_send_time = uint64_t(-1);
226}

228template <class MyState>
229void TransportSender<MyState>::add_sent_state( uint64_t the_timestamp, uint64_t num, MyState &state )
230{
231 sent_states.push_back( TimestampedState<MyState>( the_timestamp, num, state ) );
232 if ( sent_states.size() > 32 ) { /* limit on state queue */
233 typename sent_states_type::iterator last = sent_states.end();
234 for ( int i = 0; i < 16; i++ ) { last--; }
235 sent_states.erase( last ); /* erase state from middle of queue */
236 }
237}

239template <class MyState>
240void TransportSender<MyState>::send_to_receiver( const string & diff )
241{
242 uint64_t new_num;
243 if ( current_state == sent_states.back().state ) { /* previously sent */
244 new_num = sent_states.back().num;
245 } else { /* new state */
246 new_num = sent_states.back().num + 1;
247 }
248
249 /* special case for shutdown sequence */
250 if ( shutdown_in_progress ) {
251 new_num = uint64_t( -1 );
252 }
253
254 if ( new_num == sent_states.back().num ) {
255 sent_states.back().timestamp = timestamp();
256 } else {
257 add_sent_state( timestamp(), new_num, current_state );
258 }
259
260 send_in_fragments( diff, new_num ); // Can throw NetworkException
261
262 /* successfully sent, probably */
263 /* ("probably" because the FIRST size-exceeded datagram doesn't get an error) */
264 assumed_receiver_state = sent_states.end();
265 assumed_receiver_state--;
266 next_ack_time = timestamp() + ACK_INTERVAL;
267 next_send_time = uint64_t(-1);
268}

270template <class MyState>
271void TransportSender<MyState>::update_assumed_receiver_state( void )
272{
273 uint64_t now = timestamp();
274
275 /* start from what is known and give benefit of the doubt to unacknowledged states
276 transmitted recently enough ago */
277 assumed_receiver_state = sent_states.begin();
278
279 typename list< TimestampedState<MyState> >::iterator i = sent_states.begin();
280 i++;
281
282 while ( i != sent_states.end() ) {
283 assert( now >= i->timestamp );
284
285 if ( uint64_t(now - i->timestamp) < connection->timeout() + ACK_DELAY ) {
286 assumed_receiver_state = i;
287 } else {
288 return;
289 }
290
291 i++;
292 }
293}

295template <class MyState>
296void TransportSender<MyState>::rationalize_states( void )
297{
298 const MyState * known_receiver_state = &sent_states.front().state;
299
300 current_state.subtract( known_receiver_state );
301
302 for ( typename list< TimestampedState<MyState> >::reverse_iterator i = sent_states.rbegin();
303 i != sent_states.rend();
304 i++ ) {
305 i->state.subtract( known_receiver_state );
306 }
307}

309template <class MyState>
310const string TransportSender<MyState>::make_chaff( void )
311{
312 const size_t CHAFF_MAX = 16;
313 const size_t chaff_len = prng.uint8() % (CHAFF_MAX + 1);
314
315 char chaff[ CHAFF_MAX ];
316 prng.fill( chaff, chaff_len );
317 return string( chaff, chaff_len );
318}

320template <class MyState>
321void TransportSender<MyState>::send_in_fragments( const string & diff, uint64_t new_num )
322{
323 Instruction inst;
324
325 inst.set_protocol_version( MOSH_PROTOCOL_VERSION );
326 inst.set_old_num( assumed_receiver_state->num );
327 inst.set_new_num( new_num );
328 inst.set_ack_num( ack_num );
329 inst.set_throwaway_num( sent_states.front().num );
330 inst.set_diff( diff );
331 inst.set_chaff( make_chaff() );
332
333 if ( new_num == uint64_t(-1) ) {
334 shutdown_tries++;
335 }
336
337 vector<Fragment> fragments = fragmenter.make_fragments( inst, connection->get_MTU()
338 - Network::Connection::ADDED_BYTES
339 - Crypto::Session::ADDED_BYTES );
340 for ( vector<Fragment>::iterator i = fragments.begin();
341 i != fragments.end();
342 i++ ) {
343 connection->send( i->tostring() );
344
345 if ( verbose ) {
346 fprintf( stderr, "[%u] Sent [%d=>%d] id %d, frag %d ack=%d, throwaway=%d, len=%d, frame rate=%.2f, timeout=%d, srtt=%.1f\n",
347 (unsigned int)(timestamp() % 100000), (int)inst.old_num(), (int)inst.new_num(), (int)i->id, (int)i->fragment_num,
348 (int)inst.ack_num(), (int)inst.throwaway_num(), (int)i->contents.size(),
349 1000.0 / (double)send_interval(),
350 (int)connection->timeout(), connection->get_SRTT() );
351 }
352
353 }
354
355 pending_data_ack = false;
356}

358template <class MyState>
359void TransportSender<MyState>::process_acknowledgment_through( uint64_t ack_num )
360{
361 /* Ignore ack if we have culled the state it's acknowledging */
362
363 if ( sent_states.end() !=
364 find_if( sent_states.begin(), sent_states.end(),
365 bind2nd( mem_fun_ref( &TimestampedState<MyState>::num_eq ), ack_num ) ) ) {
366 sent_states.remove_if( bind2nd( mem_fun_ref( &TimestampedState<MyState>::num_lt ), ack_num ) );
367 }
368
369 assert( !sent_states.empty() );
370}

372/* give up on getting acknowledgement for shutdown */
373template <class MyState>
374bool TransportSender<MyState>::shutdown_ack_timed_out( void ) const
375{
376 if ( shutdown_in_progress ) {
377 if ( shutdown_tries >= SHUTDOWN_RETRIES ) {
378 return true;
379 } else if ( timestamp() - shutdown_start >= uint64_t( ACTIVE_RETRY_TIMEOUT ) ) {
380 return true;
381 }
382 }
383
384 return false;
385}

387/* Executed upon entry to new receiver state */
388template <class MyState>
389void TransportSender<MyState>::set_ack_num( uint64_t s_ack_num )
390{
391 ack_num = s_ack_num;

394/* Investigate diff against known receiver state instead */
395/* Mutates proposed_diff */
396template <class MyState>
397void TransportSender<MyState>::attempt_prospective_resend_optimization( string &proposed_diff )
398{
399 if ( assumed_receiver_state == sent_states.begin() ) {
400 return;
401 }
402
403 string resend_diff = current_state.diff_from( sent_states.front().state );
404
405 /* We do a prophylactic resend if it would make the diff shorter,
406 or if it would lengthen it by no more than 100 bytes and still be
407 less than 1000 bytes. */
408
409 if ( (resend_diff.size() <= proposed_diff.size())
410 || ( (resend_diff.size() < 1000)
411 && (resend_diff.size() - proposed_diff.size() < 100) ) ) {
412 assumed_receiver_state = sent_states.begin();
413 proposed_diff = resend_diff;
414 }
