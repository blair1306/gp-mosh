33#ifndef USER_HPP
34#define USER_HPP
35
36#include <deque>
37#include <list>
38#include <string>
39#include <assert.h>
40
41#include "parseraction.h"
42
43using std::deque;
44using std::list;
45using std::string;
46
47namespace Network {
48 enum UserEventType {
49 UserByteType = 0,
50 ResizeType = 1
51 };
52
53 class UserEvent
54 {
55 public:
56 UserEventType type;
57 Parser::UserByte userbyte;
58 Parser::Resize resize;
59
60 UserEvent( const Parser::UserByte & s_userbyte ) : type( UserByteType ), userbyte( s_userbyte ), resize( -1, -1 ) {}
61 UserEvent( const Parser::Resize & s_resize ) : type( ResizeType ), userbyte( 0 ), resize( s_resize ) {}
62
63 private:
64 UserEvent();
65
66 public:
67 bool operator==( const UserEvent &x ) const { return ( type == x.type ) && ( userbyte == x.userbyte ) && ( resize == x.resize ); }
68 };
69
70 class UserStream
71 {
72 private:
73 deque<UserEvent> actions;
74
75 public:
76 UserStream() : actions() {}
77
78 void push_back( const Parser::UserByte & s_userbyte ) { actions.push_back( UserEvent( s_userbyte ) ); }
79 void push_back( const Parser::Resize & s_resize ) { actions.push_back( UserEvent( s_resize ) ); }
80
81 bool empty( void ) const { return actions.empty(); }
82 size_t size( void ) const { return actions.size(); }
83 const Parser::Action &get_action( unsigned int i ) const;
84
85 /* interface for Network::Transport */
86 void subtract( const UserStream *prefix );
87 string diff_from( const UserStream &existing ) const;
88 string init_diff( void ) const { return diff_from( UserStream() ); };
89 void apply_string( const string &diff );
90 bool operator==( const UserStream &x ) const { return actions == x.actions; }
91
92 bool compare( const UserStream & ) { return false; }
93 };
94}

96#endif
