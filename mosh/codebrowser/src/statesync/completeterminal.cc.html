45string Complete::act( const string &str )
46{
47 for ( unsigned int i = 0; i < str.size(); i++ ) {
48 /* parse octet into up to three actions */
49 parser.input( str[ i ], actions );
50
51 /* apply actions to terminal and delete them */
52 for ( Actions::iterator it = actions.begin();
53 it != actions.end();
54 it++ ) {
55 Action &act = **it;
56 act.act_on_terminal( &terminal );
57 }
58 actions.clear();
59 }
60
61 return terminal.read_octets_to_host();
62}
63
64string Complete::act( const Action &act )
65{
66 /* apply action to terminal */
67 act.act_on_terminal( &terminal );
68 return terminal.read_octets_to_host();
69}

71/* interface for Network::Transport */
72string Complete::diff_from( const Complete &existing ) const
73{
74 HostBuffers::HostMessage output;
75
76 if ( existing.get_echo_ack() != get_echo_ack() ) {
77 assert( get_echo_ack() >= existing.get_echo_ack() );
78 Instruction *new_echo = output.add_instruction();
79 new_echo->MutableExtension( echoack )->set_echo_ack_num( get_echo_ack() );
80 }
81
82 if ( !(existing.get_fb() == get_fb()) ) {
83 if ( (existing.get_fb().ds.get_width() != terminal.get_fb().ds.get_width())
84 || (existing.get_fb().ds.get_height() != terminal.get_fb().ds.get_height()) ) {
85 Instruction *new_res = output.add_instruction();
86 new_res->MutableExtension( resize )->set_width( terminal.get_fb().ds.get_width() );
87 new_res->MutableExtension( resize )->set_height( terminal.get_fb().ds.get_height() );
88 }
89 string update = display.new_frame( true, existing.get_fb(), terminal.get_fb() );
90 if ( !update.empty() ) {
91 Instruction *new_inst = output.add_instruction();
92 new_inst->MutableExtension( hostbytes )->set_hoststring( update );
93 }
94 }
95
96 return output.SerializeAsString();
97}

99string Complete::init_diff( void ) const
100{
101 return diff_from( Complete( get_fb().ds.get_width(), get_fb().ds.get_height() ));

104void Complete::apply_string( const string & diff )
105{
106 HostBuffers::HostMessage input;
107 fatal_assert( input.ParseFromString( diff ) );
108
109 for ( int i = 0; i < input.instruction_size(); i++ ) {
110 if ( input.instruction( i ).HasExtension( hostbytes ) ) {
111 string terminal_to_host = act( input.instruction( i ).GetExtension( hostbytes ).hoststring() );
112 assert( terminal_to_host.empty() ); /* server never interrogates client terminal */
113 } else if ( input.instruction( i ).HasExtension( resize ) ) {
114 act( Resize( input.instruction( i ).GetExtension( resize ).width(),
115 input.instruction( i ).GetExtension( resize ).height() ) );
116 } else if ( input.instruction( i ).HasExtension( echoack ) ) {
117 uint64_t inst_echo_ack_num = input.instruction( i ).GetExtension( echoack ).echo_ack_num();
118 assert( inst_echo_ack_num >= echo_ack );
119 echo_ack = inst_echo_ack_num;
120 }
121 }
122}

124bool Complete::operator==( Complete const &x ) const
125{
126 // assert( parser == x.parser ); /* parser state is irrelevant for us */
127 return (terminal == x.terminal) && (echo_ack == x.echo_ack);
128}

130static bool old_ack(uint64_t newest_echo_ack, const pair<uint64_t, uint64_t> p)
131{
132 return p.first < newest_echo_ack;
133}

135bool Complete::set_echo_ack( uint64_t now )
136{
137 bool ret = false;
138 uint64_t newest_echo_ack = 0;
139
140 for ( input_history_type::const_iterator i = input_history.begin();
141 i != input_history.end();
142 i++ ) {
143 if ( i->second <= now - ECHO_TIMEOUT ) {
144 newest_echo_ack = i->first;
145 }
146 }
147
148 input_history.remove_if( bind1st( ptr_fun( old_ack ), newest_echo_ack ) );
149
150 if ( echo_ack != newest_echo_ack ) {
151 ret = true;
152 }
153
154 echo_ack = newest_echo_ack;
155
156 return ret;
157}

159void Complete::register_input_frame( uint64_t n, uint64_t now )
160{
161 input_history.push_back( make_pair( n, now ) );
162}

164int Complete::wait_time( uint64_t now ) const
165{
166 if ( input_history.size() < 2 ) {
167 return INT_MAX;
168 }
169
170 input_history_type::const_iterator it = input_history.begin();
171 it++;
172
173 uint64_t next_echo_ack_time = it->second + ECHO_TIMEOUT;
174 if ( next_echo_ack_time <= now ) {
175 return 0;
176 }
177 return next_echo_ack_time - now;
178}

180bool Complete::compare( const Complete &other ) const
181{
182 bool ret = false;
183 const Framebuffer &fb = terminal.get_fb();
184 const Framebuffer &other_fb = other.terminal.get_fb();
185 const int height = fb.ds.get_height();
186 const int other_height = other_fb.ds.get_height();
187 const int width = fb.ds.get_width();
188 const int other_width = other_fb.ds.get_width();
189
190 if ( height != other_height || width != other_width ) {
191 fprintf( stderr, "Framebuffer size (%dx%d, %dx%d) differs.\n", width, height, other_width, other_height );
192 return true;
193 }
194
195 for ( int y = 0; y < height; y++ ) {
196 for ( int x = 0; x < width; x++ ) {
197 if ( fb.get_cell( y, x )->compare( *other_fb.get_cell( y, x ) ) ) {
198 fprintf( stderr, "Cell (%d, %d) differs.\n", y, x );
199 ret = true;
200 }
201 }
202 }
203
204 if ( (fb.ds.get_cursor_row() != other_fb.ds.get_cursor_row())
205 || (fb.ds.get_cursor_col() != other_fb.ds.get_cursor_col()) ) {
206 fprintf( stderr, "Cursor mismatch: (%d, %d) vs. (%d, %d).\n",
207 fb.ds.get_cursor_row(), fb.ds.get_cursor_col(),
208 other_fb.ds.get_cursor_row(), other_fb.ds.get_cursor_col() );
209 ret = true;
210 }
211 /* XXX should compare other terminal state too (mouse mode, bell. etc.) */
212
213 return ret;
214}
